   153→  function voiceAppendText(text: string) {

   228→    if (state === "finishing") return

   230→    const xiaomi = sync.data.provider.find((p) => p.id === "xiaomi")

   232→      toast.show({ message: t("tui.voice.error.no_auth"), variant: "error" })
   233→      return

   235→    if (!Voice.isAvailable()) {
   236→      toast.show({ message: t("tui.voice.error.no_recorder"), variant: "error" })
   237→      return

   239→    const apiKey = xiaomi.key
   240→    const baseUrl = (xiaomi.options?.baseURL as string) || "https://api.xiaomimimo.com/v1"
   241→
   242→    const av: NonNullable<typeof activeVoice> = {
   243→      handle: undefined!,
   244→      pending: 0,
   245→      appendText: voiceAppendText,
   246→      setText: voiceSetText,
   247→      getPlainText: voiceGetPlainText,
   248→      switchAgent: voiceSwitchAgent,
   249→      submit: () => submit(),
   250→      setState: voiceSetState,
   251→      showError: (msg) => toast.show({ message: msg, variant: "error", duration: 3000 }),
   252→    }
   253→
   254→    let voiceControlChain: Promise<void> = Promise.resolve()
   255→
   256→    const handle = Voice.startStreaming({
   257→      onSegment: (segment) => {
   258→        av.pending++
   259→        av.setState("processing")
   260→
   261→        if (voiceControlEnabled()) {
   262→          voiceControlChain = voiceControlChain.then(async () => {
   263→            try {
   264→              if (!activeVoice) return
   265→              av.setState("processing")
   266→              const currentText = av.getPlainText()
   267→              const currentAgent = local.agent.current()?.name ?? ""
   268→              const availableAgents = local.agent.list().map((x) => x.name)
   269→
   270→              const ctrl = await Voice.processVoiceControl({
   271→                audio: segment.audio,
   272→                apiKey,
   273→                baseUrl,
   274→                currentText,
   275→                currentAgent,
   276→                availableAgents,
   277→                sendEnabled: voiceSendEnabled(),
   278→              })
   279→
   280→              if (ctrl) {
   281→                for (const action of ctrl.actions) {
   282→                  if (action.action === "edit") av.setText(action.text)
   283→                  else if (action.action === "send") {
   284→                    if (voiceSendEnabled() && av.getPlainText().trim()) await av.submit()
   285→                    else if (!av.getPlainText().trim()) av.showError(t("tui.voice.error.empty_send"))
   286→                  } else if (action.action === "agent") {
   287→                    av.switchAgent(action.agent)
   288→                  }
   289→                }
   290→              } else {
   291→                av.showError(t("tui.voice.error.network"))
   292→              }
   293→            } finally {
   294→              av.pending--
   295→              if (activeVoice === av) av.setState("listening")
   296→              if (!activeVoice && av.pending <= 0) av.setState("idle")
   297→            }
   298→          }).catch(() => {})
   299→        } else {
   300→          Voice.transcribeAudio({
   301→            audio: segment.audio,
   302→            apiKey,
   303→            baseUrl,
   304→          }).then((text) => {
   305→            if (text) {
   306→              if (voiceSendEnabled() && Voice.SEND_RE.test(text.replace(/[\s。.!！？?，,]+$/g, "").trim())) {
   307→                av.submit()
   308→              } else {
   309→                av.appendText(text.trim())
   310→              }
   311→            } else {
   312→              av.showError(t("tui.voice.error.network"))
   313→            }
   314→            av.pending--
   315→            if (activeVoice === av) av.setState("listening")
   316→            if (!activeVoice && av.pending <= 0) av.setState("idle")
   317→          }).catch(() => {
   318→            av.pending--
   319→            if (activeVoice === av) av.setState("listening")
   320→            if (!activeVoice && av.pending <= 0) av.setState("idle")
   321→          })
   322→        }
   323→      },
   324→      onActiveChange: (active) => {
   325→        if (active && activeVoice === av) av.setState("speaking")
   326→      },
   327→      onError: () => {
   328→        av.showError(t("tui.voice.error.no_recorder"))
   329→        activeVoice = undefined
   330→        av.setState("idle")
   331→      },
   332→    })
   333→    if (!handle) {
   334→      toast.show({ message: t("tui.voice.error.no_recorder"), variant: "error" })
   335→      return
   336→    }
   337→    av.handle = handle
   338→    activeVoice = av
   339→    setVoiceState("listening")
   340→  }

   391→  // After the agent finishes a turn, predict the user's likely next prompt and
   392→  // show it as ghost text in the empty input (accept with Tab). Only fires on
   393→  // an idle transition while the input is empty so it never clobbers typing.
   394→  let ghostRequest = 0
   395→  async function fetchGhost(sessionID: string) {
   396→    if (props.showPlaceholder === false) return
   397→    const token = ++ghostRequest
   398→    const userMessageID = lastUserMessage()?.id
   399→    const res = await sdk.client.session.predict({ sessionID }).catch(() => undefined)
   400→    const text = res?.data?.prediction?.trim()
   401→    if (!text) return
   402→    // Drop the result if anything that defined its context changed while the
   403→    // request was in flight: superseded by a newer fetch, session switched, a
   404→    // new run started, the conversation advanced, or the user began typing.
   405→    if (token !== ghostRequest) return
   406→    if (props.sessionID !== sessionID) return
   407→    if (status().type !== "idle") return
   408→    if (lastUserMessage()?.id !== userMessageID) return
   409→    if (!input || input.isDestroyed || input.plainText !== "") return
   410→    setGhost(text)

   829→  function restoreExtmarksFromParts(parts: PromptInfo["parts"]) {
   830→    input.extmarks.clear()

   839→      if (part.type === "file" && part.source?.text) {
   840→        start = part.source.text.start
   841→        end = part.source.text.end
   842→        virtualText = part.source.text.value
   843→        styleId = fileStyleId
   844→      } else if (part.type === "agent" && part.source) {
   845→        start = part.source.start
   846→        end = part.source.end
   847→        virtualText = part.source.value

   849→      } else if (part.type === "text" && part.source?.text) {
   850→        start = part.source.text.start
   851→        end = part.source.text.end
   852→        virtualText = part.source.text.value
   853→        styleId = pasteStyleId

   857→        const extmarkId = input.extmarks.create({
   858→          start,
   859→          end,
   860→          virtual: true,
   861→          styleId,
   862→          typeId: promptPartTypeId,

   865→          const newMap = new Map(map)
   866→          newMap.set(extmarkId, partIndex)

   892→                part.source.text.start = extmark.start
   893→                part.source.text.end = extmark.end

   962→  async function submit() {
   963→    setGhost("")
   964→    // IME: double-defer may fire before onContentChange flushes the last
   965→    // composed character (e.g. Korean hangul) to the store, so read
   966→    // plainText directly and sync before any downstream reads.
   967→    if (input && !input.isDestroyed && input.plainText !== store.prompt.input) {
   968→      setStore("prompt", "input", input.plainText)
   969→      syncExtmarksWithPromptParts()
   970→    }
   971→    if (props.disabled) return false
   972→    if (autocomplete?.visible) return false
   973→    if (!store.prompt.input) return false
   974→    const agent = local.agent.current()
   975→    if (!agent) return false
   976→    const trimmed = store.prompt.input.trim()
   977→    if (trimmed === "exit" || trimmed === "quit" || trimmed === ":q") {
   978→      void exit()
   979→      return true
   980→    }
   981→    const selectedModel = local.model.current()
   982→    if (!selectedModel) {
   983→      void promptModelWarning()
   984→      return false
   985→    }
   986→
   987→    const workspaceSession = props.sessionID ? sync.session.get(props.sessionID) : undefined
   988→    const workspaceID = workspaceSession?.workspaceID
   989→    const workspaceStatus = workspaceID ? (project.workspace.status(workspaceID) ?? "error") : undefined
   990→    if (props.sessionID && workspaceID && workspaceStatus !== "connected") {
   991→      dialog.replace(() => (
   992→        <DialogWorkspaceUnavailable
   993→          onRestore={() => {
   994→            dialog.replace(() => (
   995→              <DialogWorkspaceCreate
   996→                onSelect={(nextWorkspaceID) =>
   997→                  restoreWorkspaceSession({
   998→                    dialog,
   999→                    sdk,
  1000→                    sync,
  1001→                    project,
  1002→                    toast,
  1003→                    workspaceID: nextWorkspaceID,
  1004→                    sessionID: props.sessionID!,
  1005→                  })
  1006→                }
  1007→              />
  1008→            ))
  1009→          }}
  1010→        />
  1011→      ))
  1012→      return false
  1013→    }
  1014→
  1015→    let sessionID = props.sessionID
  1016→    if (sessionID == null) {
  1017→      const res = await sdk.client.session.create({ workspace: props.workspaceID })
  1018→
  1019→      if (res.error) {
  1020→        console.log("Creating a session failed:", res.error)
  1021→
  1022→        toast.show({
  1023→          message: "Creating a session failed. Open console for more details.",
  1024→          variant: "error",
  1025→        })
  1026→
  1027→        return true
  1028→      }
  1029→
  1030→      sessionID = res.data.id
  1031→    }
  1032→
  1033→    const messageID = MessageID.ascending()
  1034→    let inputText = store.prompt.input
  1035→
  1036→    // Expand pasted text inline before submitting
  1037→    const allExtmarks = input.extmarks.getAllForTypeId(promptPartTypeId)
  1038→    const sortedExtmarks = allExtmarks.sort((a: { start: number }, b: { start: number }) => b.start - a.start)
  1039→
  1040→    for (const extmark of sortedExtmarks) {
  1041→      const partIndex = store.extmarkToPartIndex.get(extmark.id)
  1042→      if (partIndex !== undefined) {
  1043→        const part = store.prompt.parts[partIndex]
  1044→        if (part?.type === "text" && part.text) {
  1045→          const before = inputText.slice(0, extmark.start)
  1046→          const after = inputText.slice(extmark.end)
  1047→          inputText = before + part.text + after
  1048→        }
  1049→      }
  1050→    }
  1051→
  1052→    // Filter out text parts (pasted content) since they're now expanded inline
  1053→    const nonTextParts = store.prompt.parts.filter((part) => part.type !== "text")
  1054→
  1055→    // Capture mode before it gets reset
  1056→    const currentMode = store.mode
  1057→    const variant = local.model.variant.current()
  1058→
  1059→    const clientSlash = inputText.startsWith("/")
  1060→      ? command.slashes().find((s) => s.display === inputText.trim())
  1061→      : undefined
  1062→
  1063→    if (store.mode === "shell") {
  1064→      void sdk.client.session.shell({
  1065→        sessionID,
  1066→        agent: agent.name,
  1067→        model: {
  1068→          providerID: selectedModel.providerID,
  1069→          modelID: selectedModel.modelID,
  1070→        },
  1071→        command: inputText,
  1072→      })
  1073→      setStore("mode", "normal")
  1074→    } else if (clientSlash) {
  1075→      clientSlash.onSelect?.()
  1076→    } else if (
  1077→      inputText.startsWith("/") &&
  1078→      iife(() => {
  1079→        const firstLine = inputText.split("\n")[0]
  1080→        const command = firstLine.split(" ")[0].slice(1)
  1081→        return sync.data.command.some((x) => x.name === command)
  1082→      })
  1083→    ) {
  1084→      // Parse command from first line, preserve multi-line content in arguments
  1085→      const firstLineEnd = inputText.indexOf("\n")
  1086→      const firstLine = firstLineEnd === -1 ? inputText : inputText.slice(0, firstLineEnd)
  1087→      const [command, ...firstLineArgs] = firstLine.split(" ")
  1088→      const restOfInput = firstLineEnd === -1 ? "" : inputText.slice(firstLineEnd + 1)
  1089→      const args = firstLineArgs.join(" ") + (restOfInput ? "\n" + restOfInput : "")
  1090→
  1091→      void sdk.client.session.command({
  1092→        sessionID,
  1093→        command: command.slice(1),
  1094→        arguments: args,
  1095→        agent: agent.name,
  1096→        model: `${selectedModel.providerID}/${selectedModel.modelID}`,
  1097→        messageID,
  1098→        variant,
  1099→        parts: nonTextParts
  1100→          .filter((x) => x.type === "file")
  1101→          .map((x) => ({
  1102→            id: PartID.ascending(),
  1103→            ...x,
  1104→          })),
  1105→      })
  1106→    } else {
  1107→      sdk.client.session
  1108→        .promptAsync({
  1109→          sessionID,
  1110→          ...selectedModel,
  1111→          messageID,
  1112→          agent: agent.name,
  1113→          model: selectedModel,
  1114→          variant,
  1115→          parts: [
  1116→            {
  1117→              id: PartID.ascending(),
  1118→              type: "text",
  1119→              text: inputText,
  1120→            },
  1121→            ...nonTextParts.map(assign),
  1122→          ],
  1123→        })
  1124→        .catch((err) => {
  1125→          toast.show({
  1126→            message: err instanceof Error ? err.message : "Failed to send message",
  1127→            variant: "error",
  1128→          })

  1137→      input: "",
  1138→      parts: [],
  1139→    })

  1147→          type: "session",
  1148→          sessionID,
  1149→        })
  1150→      }, 50)
  1151→    input.clear()
  1152→    return true
  1153→  }

  1190→  async function pastePlainText(normalizedText: string) {
  1191→    const pastedContent = normalizedText.trim()
  1192→    if (!pastedContent) return
  1193→
  1194→    const filepath = iife(() => {
  1195→      const raw = pastedContent.replace(/^['"]+|['"]+$/g, "")
  1196→      if (raw.startsWith("file://")) {
  1197→        try {
  1198→          return fileURLToPath(raw)
  1199→        } catch {}
  1200→      }
  1201→      if (process.platform === "win32") return raw
  1202→      return raw.replace(/\\(.)/g, "$1")
  1203→    })
  1204→    const isUrl = /^(https?):\/\//.test(filepath)
  1205→    if (!isUrl) {
  1206→      try {
  1207→        const mime = await Filesystem.mimeType(filepath)
  1208→        const filename = path.basename(filepath)
