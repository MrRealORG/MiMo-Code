   111→   111→/**
   112→   112→ * Deterministic JSON serialization with sorted object keys, so that two
   113→   113→ * semantically-identical tool inputs produce the same string regardless of the
   114→   114→ * order the model happened to emit the keys in. `JSON.stringify` preserves
   115→   115→ * insertion order, and models routinely re-emit the same arguments with keys in
   116→   116→ * a different order (e.g. {url,format} vs {format,url}) — without this the
   117→   117→ * signatures would differ and the repeated-step check would miss real loops.
   118→   118→ */
   119→   119→function stableStringify(value: unknown): string {
   120→   120→  if (value === null || typeof value !== "object") return JSON.stringify(value) ?? "null"
   121→   121→  if (Array.isArray(value)) return "[" + value.map(stableStringify).join(",") + "]"
   122→   122→  const keys = Object.keys(value as Record<string, unknown>).sort()
   123→   123→  return (
   124→   124→    "{" +
   125→   125→    keys.map((k) => JSON.stringify(k) + ":" + stableStringify((value as Record<string, unknown>)[k])).join(",") +
   126→   126→    "}"
   127→   127→  )
   128→   128→}
   129→   129→
   130→   130→/**
   131→   131→ * Stable signature for an assistant step's *action* — the tool calls it made
   132→   132→ * (name + key-order-independent input). Text and reasoning are excluded on
   133→   133→ * purpose: in a ReAct loop the model narrates each step in slightly different
   134→   134→ * words while taking the exact same action, and some models emit their
   135→   135→ * reasoning as plain text parts — counting either would mask the repeated
   136→   136→ * action we want to catch. Returns undefined when a step makes no tool calls
   137→   137→ * (e.g. a pure-text turn), since there is no repeated *action* to compare.
   138→   138→ */
   139→   139→function stepSignature(parts: MessageV2.Part[]): string | undefined {
   140→   140→  const segments: string[] = []
   141→   141→  for (const part of parts) {
   142→   142→    if (part.type === "tool") {
   143→   143→      segments.push("tool:" + part.tool + ":" + stableStringify(part.state.input ?? {}))
   144→   144→    }
   145→   145→  }
   146→   146→  if (segments.length === 0) return undefined
   147→   147→  return segments.join("\n")
   148→   148→}
