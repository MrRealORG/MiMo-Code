    49→// Custom merge function that concatenates array fields instead of replacing them
    50→function mergeConfigConcatArrays(target: Info, source: Info): Info {
    51→  const merged = mergeDeep(target, source)
    52→  if (target.instructions && source.instructions) {
    53→    merged.instructions = Array.from(new Set([...target.instructions, ...source.instructions]))
    54→  }
    55→  return merged
    56→}

    58→function normalizeLoadedConfig(data: unknown, source: string) {
    59→  if (!isRecord(data)) return data
    60→  const copy = { ...data }
    61→  const hadLegacy = "theme" in copy || "keybinds" in copy || "tui" in copy
    62→  if (!hadLegacy) return copy
    63→  delete copy.theme
    64→  delete copy.keybinds
    65→  delete copy.tui
    66→  log.warn("tui keys in mimocode config are deprecated; move them to tui.json", { path: source })
    67→  return copy
    68→}

    70→async function resolveLoadedPlugins<T extends { plugin?: ConfigPlugin.Spec[] }>(config: T, filepath: string) {
    71→  if (!config.plugin) return config
    72→  for (let i = 0; i < config.plugin.length; i++) {
    73→    // Normalize path-like plugin specs while we still know which config file declared them.
    74→    // This prevents `./plugin.ts` from being reinterpreted relative to some later merge location.
    75→    config.plugin[i] = await ConfigPlugin.resolvePluginSpec(config.plugin[i], filepath)
    76→  }
    77→  return config
    78→}

   409→// Schema.Struct produces readonly types by default, but the service code
   410→// below mutates Info objects directly (e.g. `config.mode = ...`). Strip the
   411→// readonly recursively so callers get the same mutable shape zod inferred.
   412→//
   413→// `Types.DeepMutable` from effect-smol would be a drop-in, but its fallback
   414→// branch `{ -readonly [K in keyof T]: ... }` collapses `unknown` to `{}`
   415→// (since `keyof unknown = never`), which widens `Record<string, unknown>`
   416→// fields like `ConfigPlugin.Options`. The local version gates on
   417→// `extends object` so `unknown` passes through.
   418→//
   419→// Tuple branch preserves `ConfigPlugin.Spec`'s `readonly [string, Options]`
   420→// shape (otherwise the general array branch widens it to an array).
   421→type DeepMutable<T> = T extends readonly [unknown, ...unknown[]]
   422→  ? { -readonly [K in keyof T]: DeepMutable<T[K]> }
   423→  : T extends readonly (infer U)[]
   424→    ? DeepMutable<U>[]
   425→    : T extends object
   426→      ? { -readonly [K in keyof T]: DeepMutable<T[K]> }
   427→      : T
   428→
   429→// The walker emits `z.object({...})` which is non-strict by default. Config
   430→// historically uses `.strict()` (additionalProperties: false in openapi.json),
   431→// so layer that on after derivation.  Re-apply the Config ref afterward
   432→// since `.strict()` strips the walker's meta annotation.
   433→export const Info = (zod(InfoSchema) as unknown as z.ZodObject<any>)
   434→  .strict()
   435→  .meta({ ref: "Config" }) as unknown as z.ZodType<DeepMutable<Schema.Schema.Type<typeof InfoSchema>>>

   464→function globalConfigFile() {
   465→  const candidates = ["mimocode.jsonc", "mimocode.json", "config.json"].map((file) =>
   466→    path.join(Global.Path.config, file),
   467→  )
   468→  for (const file of candidates) {
   469→    if (existsSync(file)) return file
   470→  }
   471→  return candidates[0]
   472→}

   474→function patchJsonc(input: string, patch: unknown, path: string[] = []): string {
   475→  if (!isRecord(patch)) {
   476→    const edits = modify(input, path, patch, {
   477→      formattingOptions: {
   478→        insertSpaces: true,
   479→        tabSize: 2,
   480→      },
   481→    })
   482→    return applyEdits(input, edits)
   483→  }
   484→
   485→  return Object.entries(patch).reduce((result, [key, value]) => {
   486→    if (value === undefined) return result
   487→    return patchJsonc(result, value, [...path, key])
   488→  }, input)
   489→}

   491→function writable(info: Info) {
   492→  const { plugin_origins: _plugin_origins, mcp_origins: _mcp_origins, ...next } = info
   493→  return next
   494→}

   622→          if (source.startsWith("http://") || source.startsWith("https://")) return "global"

   629→          source: string,
   630→          // mergePluginOrigins receives raw Specs from one config source, before provenance for this merge step
   631→          // is attached.
   632→          list: ConfigPlugin.Spec[] | undefined,
   633→          // Scope can be inferred from the source path, but some callers already know whether the config should
   634→          // behave as global or local and can pass that explicitly.
   635→          kind?: ConfigPlugin.Scope,

   637→          if (!list?.length) return

   639→          // Merge newly seen plugin origins with previously collected ones, then dedupe by plugin identity while
   640→          // keeping the winning source/scope metadata for downstream installs, writes, and diagnostics.
   641→          const plugins = ConfigPlugin.deduplicatePluginOrigins([
   642→            ...(result.plugin_origins ?? []),
   643→            ...list.map((spec) => ({ spec, source, scope: hit })),

   991→        const merged = mergeDeep(writable(existing), writable(config))

   993→        next = merged
   994→      } else {
   995→        const updated = patchJsonc(before, writable(config))
   996→        next = ConfigParse.schema(Info, ConfigParse.jsonc(updated, file), file)
   997→        yield* fs.writeFileString(file, updated).pipe(Effect.orDie)
   998→      }
   999→
  1000→      yield* invalidate()
  1001→      return next
  1002→    })

  1004→    return Service.of({
  1005→      get,
  1006→      getGlobal,
  1007→      getConsoleState,
