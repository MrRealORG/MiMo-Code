    54→    54→function wrapSSE(res: Response, ms: number, ctl: AbortController) {
    55→    55→  if (typeof ms !== "number" || ms <= 0) return res
    56→    56→  if (!res.body) return res
    57→    57→  if (!res.headers.get("content-type")?.includes("text/event-stream")) return res
    58→    58→
    59→    59→  const reader = res.body.getReader()
    60→    60→  const body = new ReadableStream<Uint8Array>({
    61→    61→    async pull(ctrl) {
    62→    62→      const part = await new Promise<Awaited<ReturnType<typeof reader.read>>>((resolve, reject) => {
    63→    63→        const id = setTimeout(() => {
    64→    64→          const err = new Error("SSE read timed out")
    65→    65→          ctl.abort(err)
    66→    66→          void reader.cancel(err)
    67→    67→          reject(err)
    68→    68→        }, ms)
    69→    69→
    70→    70→        reader.read().then(
    71→    71→          (part) => {
    72→    72→            clearTimeout(id)
    73→    73→            resolve(part)
    74→    74→          },
    75→    75→          (err) => {
    76→    76→            clearTimeout(id)
    77→    77→            reject(err)
    78→    78→          },
    79→    79→        )
    80→    80→      })
    81→    81→
    82→    82→      if (part.done) {
    83→    83→        ctrl.close()
    84→    84→        return
    85→    85→      }
    86→    86→
    87→    87→      ctrl.enqueue(part.value)
    88→    88→    },
    89→    89→    async cancel(reason) {
    90→    90→      ctl.abort(reason)
    91→    91→      await reader.cancel(reason)
    92→    92→    },
    93→    93→  })
    94→    94→
    95→    95→  return new Response(body, {
    96→    96→    headers: new Headers(res.headers),
    97→    97→    status: res.status,
    98→    98→    statusText: res.statusText,
    99→    99→  })
   100→   100→}
