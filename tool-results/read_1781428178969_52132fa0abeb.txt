    12→function mimeToModality(mime: string): Modality | undefined {
    13→  if (mime.startsWith("image/")) return "image"
    14→  if (mime.startsWith("audio/")) return "audio"
    15→  if (mime.startsWith("video/")) return "video"
    16→  if (mime === "application/pdf") return "pdf"
    17→  return undefined
    18→}

    33→  if (id.includes("gpt") || apiID.includes("gpt")) return true
    34→  return model.capabilities.input.image

    66→function normalizeMessages(
    67→  msgs: ModelMessage[],
    68→  model: Provider.Model,
    69→  _options: Record<string, unknown>,
    70→): ModelMessage[] {
    71→  // Anthropic rejects messages with empty content - filter out empty string messages
    72→  // and remove empty text/reasoning parts from array content
    73→  if (model.api.npm === "@ai-sdk/anthropic" || model.api.npm === "@ai-sdk/amazon-bedrock") {
    74→    msgs = msgs
    75→      .map((msg) => {
    76→        if (typeof msg.content === "string") {
    77→          if (msg.content === "") return undefined
    78→          return msg
    79→        }
    80→        if (!Array.isArray(msg.content)) return msg
    81→        const filtered = msg.content.filter((part) => {
    82→          if (part.type === "text" || part.type === "reasoning") {
    83→            return part.text !== ""
    84→          }
    85→          return true
    86→        })
    87→        if (filtered.length === 0) return undefined
    88→        return { ...msg, content: filtered }
    89→      })
    90→      .filter((msg): msg is ModelMessage => msg !== undefined && msg.content !== "")
    91→  }
    92→
    93→  if (model.api.id.includes("claude")) {
    94→    const scrub = (id: string) => id.replace(/[^a-zA-Z0-9_-]/g, "_")
    95→    msgs = msgs.map((msg) => {
    96→      if (msg.role === "assistant" && Array.isArray(msg.content)) {
    97→        return {
    98→          ...msg,
    99→          content: msg.content.map((part) => {
   100→            if (part.type === "tool-call" || part.type === "tool-result") {
   101→              return { ...part, toolCallId: scrub(part.toolCallId) }
   102→            }
   103→            return part
   104→          }),
   105→        }
   106→      }
   107→      if (msg.role === "tool" && Array.isArray(msg.content)) {
   108→        return {
   109→          ...msg,
   110→          content: msg.content.map((part) => {
   111→            if (part.type === "tool-result") {
   112→              return { ...part, toolCallId: scrub(part.toolCallId) }
   113→            }
   114→            return part
   115→          }),
   116→        }
   117→      }
   118→      return msg
   119→    })
   120→  }
   121→  if (["@ai-sdk/anthropic", "@ai-sdk/google-vertex/anthropic"].includes(model.api.npm)) {
   122→    // Anthropic rejects assistant turns where tool_use blocks are followed by non-tool
   123→    // content, e.g. [tool_use, tool_use, text], with:
   124→    // `tool_use` ids were found without `tool_result` blocks immediately after...
   125→    //
   126→    // Reorder that invalid shape into [text] + [tool_use, tool_use]. Consecutive
   127→    // assistant messages are later merged by the provider/SDK, so preserving the
   128→    // original [tool_use...] then [text] order still produces the invalid payload.
   129→    //
   130→    // The root cause appears to be somewhere upstream where the stream is originally
   131→    // processed. We were unable to locate an exact narrower reproduction elsewhere,
   132→    // so we keep this transform in place for the time being.
   133→    msgs = msgs.flatMap((msg) => {
   134→      if (msg.role !== "assistant" || !Array.isArray(msg.content)) return [msg]
   135→
   136→      const parts = msg.content
   137→      const first = parts.findIndex((part) => part.type === "tool-call")
   138→      if (first === -1) return [msg]
   139→      if (!parts.slice(first).some((part) => part.type !== "tool-call")) return [msg]
   140→      return [
   141→        { ...msg, content: parts.filter((part) => part.type !== "tool-call") },
   142→        { ...msg, content: parts.filter((part) => part.type === "tool-call") },
   143→      ]
   144→    })
   145→  }
   146→  if (
   147→    model.providerID === "mistral" ||
   148→    model.api.id.toLowerCase().includes("mistral") ||
   149→    model.api.id.toLocaleLowerCase().includes("devstral")
   150→  ) {
   151→    const scrub = (id: string) => {
   152→      return id
   153→        .replace(/[^a-zA-Z0-9]/g, "") // Remove non-alphanumeric characters
   154→        .substring(0, 9) // Take first 9 characters
   155→        .padEnd(9, "0") // Pad with zeros if less than 9 characters
   156→    }
   157→    const result: ModelMessage[] = []
   158→    for (let i = 0; i < msgs.length; i++) {
   159→      const msg = msgs[i]
   160→      const nextMsg = msgs[i + 1]
   161→
   162→      if (msg.role === "assistant" && Array.isArray(msg.content)) {
   163→        msg.content = msg.content.map((part) => {
   164→          if (part.type === "tool-call" || part.type === "tool-result") {
   165→            return { ...part, toolCallId: scrub(part.toolCallId) }
   166→          }
   167→          return part
   168→        })
   169→      }
   170→      if (msg.role === "tool" && Array.isArray(msg.content)) {
   171→        msg.content = msg.content.map((part) => {
   172→          if (part.type === "tool-result") {
   173→            return { ...part, toolCallId: scrub(part.toolCallId) }
   174→          }
   175→          return part
   176→        })
   177→      }
   178→      result.push(msg)
   179→
   180→      // Fix message sequence: tool messages cannot be followed by user messages
   181→      if (msg.role === "tool" && nextMsg?.role === "user") {
   182→        result.push({
   183→          role: "assistant",
   184→          content: [
   185→            {
   186→              type: "text",
   187→              text: "Done.",
   188→            },
   189→          ],
   190→        })
   191→      }
   192→    }
   193→    return result
   194→  }
   195→
   196→  if (typeof model.capabilities.interleaved === "object" && model.capabilities.interleaved.field) {
   197→    const field = model.capabilities.interleaved.field
   198→    return msgs.map((msg) => {
   199→      if (msg.role === "assistant" && Array.isArray(msg.content)) {
   200→        const reasoningParts = msg.content.filter((part: any) => part.type === "reasoning")
   201→        const reasoningText = reasoningParts.map((part: any) => part.text).join("")
   202→
   203→        // Filter out reasoning parts from content
   204→        const filteredContent = msg.content.filter((part: any) => part.type !== "reasoning")
   205→
   206→        // Include reasoning_content | reasoning_details directly on the message for all assistant messages
   207→        if (reasoningText) {
   208→          return {
   209→            ...msg,
   210→            content: filteredContent,
   211→            providerOptions: {
   212→              ...msg.providerOptions,
   213→              openaiCompatible: {
   214→                ...msg.providerOptions?.openaiCompatible,
   215→                [field]: reasoningText,
   216→              },
   217→            },
   218→          }
   219→        }
   220→
   221→        return {
   222→          ...msg,
   223→          content: filteredContent,
   224→        }
   225→      }
   226→
   227→      return msg
   228→    })
   229→  }
   230→
   231→  return msgs
   232→}

   234→// Determines whether a model's provider respects inline cache_control / cachePoint
   235→// markers. Pure name matching (model.api.id.includes("claude")) is fragile — a Claude
   236→// model behind an OpenAI-compatible proxy gets matched but markers are silently dropped.
   237→// See upstream opencode#26786.
   238→function supportsCacheMarkers(model: Provider.Model): boolean {
   239→  // Anthropic-only SDKs — always support inline markers
   240→  if (model.api.npm === "@ai-sdk/anthropic" || model.api.npm === "@ai-sdk/google-vertex/anthropic") return true
   241→  if (model.providerID === "anthropic" || model.providerID === "google-vertex-anthropic") return true
   242→  // Bedrock cachePoint is a Converse API feature, works across model families
   243→  if (model.api.npm === "@ai-sdk/amazon-bedrock") return true
   244→  // Multi-model providers: only Anthropic/Claude models support cache markers
   245→  if (
   246→    model.api.npm === "@openrouter/ai-sdk-provider" ||
   247→    model.api.npm === "@ai-sdk/github-copilot" ||
   248→    model.api.npm === "@ai-sdk/alibaba"
   249→  ) {
   250→    return (
   251→      model.api.id.includes("claude") ||
   252→      model.api.id.includes("anthropic") ||
   253→      model.id.includes("claude") ||
   254→      model.id.includes("anthropic")
   255→    )
   256→  }
   257→  return false
   258→}

   260→function applyCaching(msgs: ModelMessage[], model: Provider.Model): ModelMessage[] {
   261→  // Only Anthropic and OpenRouter expose a cache-control TTL in their AI SDK;
   262→  // the other providers ignore an unknown `ttl` field, so we only thread it
   263→  // into those two branches. Default (unset) stays the provider 5m default.
   264→  const ttl = model.cachePromptTTL === "1h" ? { ttl: "1h" as const } : {}
   265→  const providerOptions = {
   266→    anthropic: {
   267→      cacheControl: { type: "ephemeral", ...ttl },
   268→    },
   269→    openrouter: {
   270→      cacheControl: { type: "ephemeral", ...ttl },
   271→    },
   272→    bedrock: {
   273→      cachePoint: { type: "default" },
   274→    },
   275→    openaiCompatible: {
   276→      cache_control: { type: "ephemeral" },
   277→    },
   278→    copilot: {
   279→      copilot_cache_control: { type: "ephemeral" },
   280→    },
   281→    alibaba: {
   282→      cacheControl: { type: "ephemeral" },
   283→    },
   284→  }
   285→
   286→  // Strategy: place cache breakpoints at stable prefix boundaries (max 4 allowed by Anthropic)
   287→  // 1. Last system message — system prompt never changes
   288→  // 2. Midpoint of conversation history — long prefix second-level cache
   289→  // 3. Message before the last user message — stable history boundary
   290→  const targets: ModelMessage[] = []
   291→
   292→  const systemMsgs = msgs.filter((msg) => msg.role === "system")
   293→  if (systemMsgs.length > 0) targets.push(systemMsgs[systemMsgs.length - 1])
   294→
   295→  const nonSystem = msgs.filter((msg) => msg.role !== "system")
   296→  const lastUserIdx = nonSystem.findLastIndex((msg) => msg.role === "user")
   297→  if (lastUserIdx >= 1) {
   298→    targets.push(nonSystem[lastUserIdx - 1])
   299→    const midpoint = Math.floor(lastUserIdx / 2)
   300→    if (midpoint > 0 && midpoint < lastUserIdx - 1) targets.push(nonSystem[midpoint])
   301→  } else if (lastUserIdx === 0) {
   302→    targets.push(nonSystem[0])
   303→  }
   304→
   305→  for (const msg of unique(targets)) {
   306→    const useMessageLevelOptions =
   307→      model.providerID === "anthropic" ||
   308→      model.providerID.includes("bedrock") ||
   309→      model.api.npm === "@ai-sdk/amazon-bedrock"
   310→    const shouldUseContentOptions = !useMessageLevelOptions && Array.isArray(msg.content) && msg.content.length > 0
   311→
   312→    if (shouldUseContentOptions) {
   313→      const lastContent = msg.content[msg.content.length - 1]
   314→      if (
   315→        lastContent &&
   316→        typeof lastContent === "object" &&
   317→        lastContent.type !== "tool-approval-request" &&
   318→        lastContent.type !== "tool-approval-response"
   319→      ) {
   320→        lastContent.providerOptions = mergeDeep(lastContent.providerOptions ?? {}, providerOptions)
   321→        continue
   322→      }
   323→    }
   324→
   325→    msg.providerOptions = mergeDeep(msg.providerOptions ?? {}, providerOptions)
   326→  }
   327→
   328→  return msgs
   329→}

   331→function unsupportedParts(msgs: ModelMessage[], model: Provider.Model): ModelMessage[] {
   332→  return msgs.map((msg) => {
   333→    if (msg.role !== "user" || !Array.isArray(msg.content)) return msg
   334→
   335→    const filtered = msg.content.map((part) => {
   336→      if (part.type !== "file" && part.type !== "image") return part
   337→
   338→      // Check for empty base64 image data
   339→      if (part.type === "image") {
   340→        const imageStr = String(part.image)
   341→        if (imageStr.startsWith("data:")) {
   342→          const match = imageStr.match(/^data:([^;]+);base64,(.*)$/)
   343→          if (match && (!match[2] || match[2].length === 0)) {
   344→            return {
   345→              type: "text" as const,
   346→              text: "ERROR: Image file is empty or corrupted. Please provide a valid image.",
   347→            }
   348→          }
   349→        }
   350→      }
   351→
   352→      const mime = part.type === "image" ? String(part.image).split(";")[0].replace("data:", "") : part.mediaType
   353→      const filename = part.type === "file" ? part.filename : undefined
   354→      const modality = mimeToModality(mime)
   355→      if (!modality) return part
   356→      const supported = modality === "image" ? supportsImageInput(model) : model.capabilities.input[modality]
   357→      if (supported) return part
   358→
   359→      const name = filename ? `"${filename}"` : modality
   360→      return {
   361→        type: "text" as const,
   362→        text: `ERROR: Cannot read ${name} (this model does not support ${modality} input). Inform the user.`,
   363→      }
   364→    })
   365→
   366→    return { ...msg, content: filtered }
   367→  })
   368→}

   370→// Returns the decoded byte size of a base64 data URL, or undefined for inputs
   371→// that aren't data URLs (remote URLs, raw binary) and therefore can't be sized.
   372→function imageByteSize(image: string): number | undefined {
   373→  if (!image.startsWith("data:")) return undefined
   374→  const base64 = image.slice(image.indexOf(",") + 1)
   375→  if (!base64) return 0
   376→  const padding = base64.endsWith("==") ? 2 : base64.endsWith("=") ? 1 : 0
   377→  return Math.floor((base64.length * 3) / 4) - padding
   378→}

   380→function limitImages(msgs: ModelMessage[]): ModelMessage[] {
   381→  const maxImages = Flag.MIMOCODE_MAX_PROMPT_IMAGES
   382→  const maxSize = Flag.MIMOCODE_MAX_PROMPT_IMAGE_SIZE
   383→  if (maxImages === undefined && maxSize === undefined) return msgs
   384→
   385→  const total = msgs.reduce(
   386→    (sum, msg) =>
   387→      msg.role === "user" && Array.isArray(msg.content)
   388→        ? sum + msg.content.filter((part) => part.type === "image").length
   389→        : sum,
   390→    0,
   391→  )
   392→  // Drop the oldest excess images so the most recent ones reach the model.
   393→  let toDrop = maxImages === undefined ? 0 : Math.max(0, total - maxImages)
   394→
   395→  return msgs.map((msg) => {
   396→    if (msg.role !== "user" || !Array.isArray(msg.content)) return msg
   397→    const content = msg.content.map((part) => {
   398→      if (part.type !== "image") return part
   399→      if (toDrop > 0) {
   400→        toDrop--
   401→        return { type: "text" as const, text: `[Image omitted: exceeds the configured limit of ${maxImages} prompt image(s).]` }
   402→      }
   403→      if (maxSize !== undefined) {
   404→        const size = imageByteSize(String(part.image))
   405→        if (size !== undefined && size > maxSize) {
   406→          return { type: "text" as const, text: `[Image omitted: exceeds the configured ${maxSize}-byte prompt image size limit.]` }
   407→        }
   408→      }
   409→      return part
   410→    })
   411→    return { ...msg, content }
   412→  })
   413→}

   415→export function message(msgs: ModelMessage[], model: Provider.Model, options: Record<string, unknown>) {
   416→  msgs = unsupportedParts(msgs, model)
   417→  msgs = limitImages(msgs)
   418→  msgs = normalizeMessages(msgs, model, options)
   419→  if (supportsCacheMarkers(model)) {
   420→    msgs = applyCaching(msgs, model)
   421→  }
   422→
   423→  // Remap providerOptions keys from stored providerID to expected SDK key
   424→  const key = sdkKey(model.api.npm)
   425→  if (key && key !== model.providerID) {
   426→    const remap = (opts: Record<string, any> | undefined) => {
   427→      if (!opts) return opts
   428→      if (!(model.providerID in opts)) return opts
   429→      const result = { ...opts }
   430→      result[key] = result[model.providerID]
   431→      delete result[model.providerID]
   432→      return result
   433→    }
   434→
   435→    msgs = msgs.map((msg) => {
   436→      if (!Array.isArray(msg.content)) return { ...msg, providerOptions: remap(msg.providerOptions) }
   437→      return {
   438→        ...msg,
   439→        providerOptions: remap(msg.providerOptions),
   440→        content: msg.content.map((part) => {
   441→          if (part.type === "tool-approval-request" || part.type === "tool-approval-response") {
   442→            return { ...part }
   443→          }
   444→          return { ...part, providerOptions: remap(part.providerOptions) }
   445→        }),
   446→      } as typeof msg
   447→    })
   448→  }
   449→
   450→  return msgs
   451→}

   504→export function variants(model: Provider.Model): Record<string, Record<string, any>> {
   505→  if (!model.capabilities.reasoning) return {}
   506→
   507→  const id = model.id.toLowerCase()
   508→  const adaptiveEfforts = anthropicAdaptiveEfforts(model.api.id)
   509→  if (
   510→    id.includes("deepseek") ||
   511→    id.includes("minimax") ||
   512→    id.includes("glm") ||
   513→    id.includes("mistral") ||
   514→    id.includes("kimi") ||
   515→    id.includes("k2p5") ||
   516→    id.includes("qwen") ||
   517→    id.includes("big-pickle")
   518→  )
   519→    return {}
   520→
   521→  // see: https://docs.x.ai/docs/guides/reasoning#control-how-hard-the-model-thinks
   522→  if (id.includes("grok") && id.includes("grok-3-mini")) {
   523→    if (model.api.npm === "@openrouter/ai-sdk-provider") {
   524→      return {
   525→        low: { reasoning: { effort: "low" } },
   526→        high: { reasoning: { effort: "high" } },
   527→      }
   528→    }
   529→    return {
   530→      low: { reasoningEffort: "low" },
   531→      high: { reasoningEffort: "high" },
   532→    }
   533→  }
   534→  if (id.includes("grok")) return {}
   535→

   894→export function options(input: {
   895→  model: Provider.Model
   896→  sessionID: string
   897→  providerOptions?: Record<string, any>
   898→}): Record<string, any> {
   899→  const result: Record<string, any> = {}
   900→
   901→  // openai and providers using openai package should set store to false by default.
   902→  if (
   903→    input.model.providerID === "openai" ||
   904→    input.model.api.npm === "@ai-sdk/openai" ||
   905→    input.model.api.npm === "@ai-sdk/github-copilot"
   906→  ) {
   907→    result["store"] = false
   908→  }
   909→
   910→  if (input.model.api.npm === "@ai-sdk/azure") {
   911→    result["store"] = true
   912→    result["promptCacheKey"] = input.sessionID
   913→  }
   914→
   915→  if (input.model.api.npm === "@openrouter/ai-sdk-provider" || input.model.api.npm === "@llmgateway/ai-sdk-provider") {
   916→    result["usage"] = {
   917→      include: true,
   918→    }
   919→    if (input.model.api.id.includes("gemini-3")) {
   920→      result["reasoning"] = { effort: "high" }
   921→    }
   922→  }
   923→
   924→  if (
   925→    input.model.providerID === "baseten" ||
   926→    (input.model.providerID === "opencode" && ["kimi-k2-thinking", "glm-4.6"].includes(input.model.api.id))
   927→  ) {
   928→    result["chat_template_args"] = { enable_thinking: true }
   929→  }
   930→
   931→  if (
   932→    ["zai", "zhipuai"].some((id) => input.model.providerID.includes(id)) &&
   933→    input.model.api.npm === "@ai-sdk/openai-compatible"
   934→  ) {
   935→    result["thinking"] = {
   936→      type: "enabled",
   937→      clear_thinking: false,
   938→    }
   939→  }
   940→
   941→  if (input.model.providerID === "openai" || input.providerOptions?.setCacheKey) {
   942→    result["promptCacheKey"] = input.sessionID
   943→  }
   944→
   945→  if (input.model.api.npm === "@ai-sdk/google" || input.model.api.npm === "@ai-sdk/google-vertex") {
   946→    if (input.model.capabilities.reasoning) {
   947→      result["thinkingConfig"] = {
   948→        includeThoughts: true,
   949→      }
   950→      if (input.model.api.id.includes("gemini-3")) {
   951→        result["thinkingConfig"]["thinkingLevel"] = "high"
   952→      }
   953→    }
   954→  }
   955→
   956→  // Enable thinking by default for kimi-k2.5/k2p5 models using anthropic SDK
   957→  const modelId = input.model.api.id.toLowerCase()
   958→  if (
   959→    (input.model.api.npm === "@ai-sdk/anthropic" || input.model.api.npm === "@ai-sdk/google-vertex/anthropic") &&
   960→    (modelId.includes("k2p5") || modelId.includes("kimi-k2.5") || modelId.includes("kimi-k2p5"))
   961→  ) {
   962→    result["thinking"] = {
   963→      type: "enabled",
   964→      budgetTokens: Math.min(16_000, Math.floor(input.model.limit.output / 2 - 1)),
   965→    }
   966→  }
   967→
   968→  // Enable thinking for reasoning models on alibaba-cn (DashScope).
   969→  // DashScope's OpenAI-compatible API requires `enable_thinking: true` in the request body
   970→  // to return reasoning_content. Without it, models like kimi-k2.5, qwen-plus, qwen3, qwq,
   971→  // deepseek-r1, etc. never output thinking/reasoning tokens.
   972→  // Note: kimi-k2-thinking is excluded as it returns reasoning_content by default.
   973→  if (
   974→    input.model.providerID === "alibaba-cn" &&
   975→    input.model.capabilities.reasoning &&
   976→    input.model.api.npm === "@ai-sdk/openai-compatible" &&
   977→    !modelId.includes("kimi-k2-thinking")
   978→  ) {
   979→    result["enable_thinking"] = true
   980→  }
   981→
   982→  if (input.model.api.id.includes("gpt-5") && !input.model.api.id.includes("gpt-5-chat")) {
   983→    if (!input.model.api.id.includes("gpt-5-pro")) {
   984→      result["reasoningEffort"] = "medium"
   985→      // Only inject reasoningSummary for providers that support it natively.
   986→      // @ai-sdk/openai-compatible proxies (e.g. LiteLLM) do not understand this
   987→      // parameter and return "Unknown parameter: 'reasoningSummary'".
   988→      if (
   989→        input.model.api.npm === "@ai-sdk/openai" ||
   990→        input.model.api.npm === "@ai-sdk/azure" ||
   991→        input.model.api.npm === "@ai-sdk/github-copilot"
   992→      ) {
   993→        result["reasoningSummary"] = "auto"
   994→      }
   995→    }
   996→
   997→    // Only set textVerbosity for non-chat gpt-5.x models
   998→    // Chat models (e.g. gpt-5.2-chat-latest) only support "medium" verbosity
   999→    if (
  1000→      input.model.api.id.includes("gpt-5.") &&
  1001→      !input.model.api.id.includes("codex") &&
  1002→      !input.model.api.id.includes("-chat") &&
  1003→      input.model.providerID !== "azure"
  1004→    ) {
  1005→      result["textVerbosity"] = "low"
  1006→    }
  1007→
  1008→    if (input.model.providerID.startsWith("opencode")) {
  1009→      result["promptCacheKey"] = input.sessionID
  1010→      result["include"] = ["reasoning.encrypted_content"]
  1011→      result["reasoningSummary"] = "auto"
  1012→    }
  1013→  }
  1014→
  1015→  if (input.model.providerID === "venice") {
  1016→    result["promptCacheKey"] = input.sessionID
  1017→  }
  1018→
  1019→  if (input.model.providerID === "openrouter") {
  1020→    result["prompt_cache_key"] = input.sessionID
  1021→  }
  1022→  if (input.model.api.npm === "@ai-sdk/gateway") {
  1023→    result["gateway"] = {
  1024→      caching: "auto",
  1025→    }
  1026→  }
  1027→
  1028→  return result
  1029→}

  1120→// Flatten a root-level `anyOf` / `oneOf` (typically from `z.discriminatedUnion`)
  1121→// into a single `type: "object"` schema with all variant properties merged at
  1122→// the root. The discriminator key becomes an `enum`, and per-variant required
  1123→// fields are encoded as a textual hint on the discriminator's description.
  1124→//
  1125→// OpenAI's function-calling validator rejects oneOf/anyOf/allOf/enum/not at the
  1126→// top level outright, so this is the only shape that gets through. We retain
  1127→// `additionalProperties: false` to keep the model from inventing fields, and
  1128→// rely on zod's runtime parse (still using the original discriminated union)
  1129→// to enforce per-action required fields strictly.
  1130→function flattenDiscriminatedUnion(schema: JSONSchema.BaseSchema | JSONSchema7): JSONSchema7 {
  1131→  const root = schema as Record<string, any>
  1132→  const variants = (root.anyOf ?? root.oneOf) as Array<Record<string, any>> | undefined
  1133→  if (!variants?.length) return schema as JSONSchema7
  1134→
  1135→  // Find the discriminator: a property whose `const` differs across all variants.
  1136→  const constByKey = new Map<string, Set<unknown>>()
  1137→  for (const v of variants) {
  1138→    if (!v.properties) continue
  1139→    for (const [key, prop] of Object.entries(v.properties as Record<string, any>)) {
  1140→      if (prop && typeof prop === "object" && "const" in prop) {
  1141→        if (!constByKey.has(key)) constByKey.set(key, new Set())
  1142→        constByKey.get(key)!.add(prop.const)
  1143→      }
  1144→    }
  1145→  }
  1146→  let discriminator: string | undefined
  1147→  for (const [key, values] of constByKey) {
  1148→    if (values.size === variants.length) {
  1149→      discriminator = key
  1150→      break
  1151→    }
  1152→  }
  1153→
  1154→  // Merge non-discriminator properties from every variant. Track which variants
  1155→  // each property appeared in so the description can tell the model
  1156→  // "(only when action='X'|'Y')" — flat schemas tempt some models (notably
  1157→  // gpt-5.5) to fill every property regardless of which action they chose.
  1158→  const properties: Record<string, any> = {}
  1159→  const propertyOwners: Record<string, unknown[]> = {}
  1160→  for (const v of variants) {
  1161→    if (!v.properties) continue
  1162→    const variantValue = discriminator
  1163→      ? (v.properties as Record<string, any>)[discriminator]?.const
  1164→      : undefined
  1165→    for (const [key, prop] of Object.entries(v.properties as Record<string, any>)) {
  1166→      if (key === discriminator) continue
  1167→      if (!(key in properties)) properties[key] = prop
  1168→      if (variantValue !== undefined) {
  1169→        if (!propertyOwners[key]) propertyOwners[key] = []
  1170→        propertyOwners[key].push(variantValue)
  1171→      }
  1172→    }
  1173→  }
  1174→  if (discriminator) {
  1175→    for (const [key, owners] of Object.entries(propertyOwners)) {
  1176→      if (owners.length === variants.length) continue // present in every variant — no annotation
  1177→      const tag = `(only when ${discriminator}=${owners.map((o) => JSON.stringify(o)).join("|")})`
  1178→      const original = (properties[key] as Record<string, any>).description as string | undefined
  1179→      properties[key] = {
  1180→        ...properties[key],
  1181→        description: original ? `${tag} ${original}` : tag,
  1182→      }
  1183→    }
  1184→  }
  1185→
  1186→  // Discriminator becomes an enum with per-variant required fields hinted in
  1187→  // its description. Without this hint the model only sees a flat bag of
  1188→  // optional fields and forgets what to provide for each action.
  1189→  if (discriminator) {
  1190→    const proto = (variants[0].properties as Record<string, any>)[discriminator]
  1191→    const enumValues = variants.map((v) => (v.properties as Record<string, any>)[discriminator!]?.const)
  1192→    const baseDescription = (proto?.description as string | undefined) ?? ""
  1193→    const hints = variants
  1194→      .map((v) => {
  1195→        const value = (v.properties as Record<string, any>)[discriminator!]?.const
  1196→        const required = ((v.required as string[] | undefined) ?? []).filter((r) => r !== discriminator)
  1197→        return required.length > 0 ? `${value}: requires ${required.join(", ")}` : `${value}: no extra required fields`
  1198→      })
  1199→      .join("; ")
  1200→    properties[discriminator] = {
  1201→      type: "string",
  1202→      enum: enumValues,
  1203→      description: baseDescription ? `${baseDescription}\n\nPer-${discriminator}: ${hints}.` : `Per-${discriminator}: ${hints}.`,
  1204→    }
  1205→  }
  1206→
  1207→  return {
  1208→    type: "object",
  1209→    properties,
  1210→    required: discriminator ? [discriminator] : [],
  1211→    additionalProperties: false,
  1212→  } as JSONSchema7
  1213→}
