    12→    12→function mimeToModality(mime: string): Modality | undefined {
    13→    13→  if (mime.startsWith("image/")) return "image"
    14→    14→  if (mime.startsWith("audio/")) return "audio"
    15→    15→  if (mime.startsWith("video/")) return "video"
    16→    16→  if (mime === "application/pdf") return "pdf"
    17→    17→  return undefined
    18→    18→}

    23→    23→function supportsImageInput(model: Provider.Model): boolean {
    24→    24→  if (model.providerID === "mimo" || model.providerID === "xiaomi") {
    25→    25→    const id = model.id.toLowerCase()
    26→    26→    if (id.includes("v2.5-pro")) return false
    27→    27→    if (id === "mimo-auto" || id.includes("v2.5")) return true
    28→    28→  }

    30→    30→  const id = model.id.toLowerCase()
    31→    31→  const apiID = model.api.id.toLowerCase()
    32→    32→  if (id.includes("claude") || apiID.includes("claude") || model.providerID === "anthropic") return true
    33→    33→  if (id.includes("gpt") || apiID.includes("gpt")) return true
    34→    34→  return model.capabilities.input.image
    35→    35→}

    66→    66→function normalizeMessages(
    67→    67→  msgs: ModelMessage[],
    68→    68→  model: Provider.Model,
    69→    69→  _options: Record<string, unknown>,
    70→    70→): ModelMessage[] {
    71→    71→  // Anthropic rejects messages with empty content - filter out empty string messages
    72→    72→  // and remove empty text/reasoning parts from array content
    73→    73→  if (model.api.npm === "@ai-sdk/anthropic" || model.api.npm === "@ai-sdk/amazon-bedrock") {
    74→    74→    msgs = msgs
    75→    75→      .map((msg) => {
    76→    76→        if (typeof msg.content === "string") {
    77→    77→          if (msg.content === "") return undefined
    78→    78→          return msg
    79→    79→        }
    80→    80→        if (!Array.isArray(msg.content)) return msg
    81→    81→        const filtered = msg.content.filter((part) => {
    82→    82→          if (part.type === "text" || part.type === "reasoning") {
    83→    83→            return part.text !== ""
    84→    84→          }
    85→    85→          return true
    86→    86→        })
    87→    87→        if (filtered.length === 0) return undefined
    88→    88→        return { ...msg, content: filtered }
    89→    89→      })
    90→    90→      .filter((msg): msg is ModelMessage => msg !== undefined && msg.content !== "")
    91→    91→  }
    92→    92→
    93→    93→  if (model.api.id.includes("claude")) {
    94→    94→    const scrub = (id: string) => id.replace(/[^a-zA-Z0-9_-]/g, "_")
    95→    95→    msgs = msgs.map((msg) => {
    96→    96→      if (msg.role === "assistant" && Array.isArray(msg.content)) {
    97→    97→        return {
    98→    98→          ...msg,
    99→    99→          content: msg.content.map((part) => {
   100→   100→            if (part.type === "tool-call" || part.type === "tool-result") {
   101→   101→              return { ...part, toolCallId: scrub(part.toolCallId) }
   102→   102→            }
   103→   103→            return part
   104→   104→          }),
   105→   105→        }
   106→   106→      }
   107→   107→      if (msg.role === "tool" && Array.isArray(msg.content)) {
   108→   108→        return {
   109→   109→          ...msg,
   110→   110→          content: msg.content.map((part) => {
   111→   111→            if (part.type === "tool-result") {
   112→   112→              return { ...part, toolCallId: scrub(part.toolCallId) }
   113→   113→            }
   114→   114→            return part
   115→   115→          }),
   116→   116→        }
   117→   117→      }
   118→   118→      return msg
   119→   119→    })
   120→   120→  }
   121→   121→  if (["@ai-sdk/anthropic", "@ai-sdk/google-vertex/anthropic"].includes(model.api.npm)) {
   122→   122→    // Anthropic rejects assistant turns where tool_use blocks are followed by non-tool
   123→   123→    // content, e.g. [tool_use, tool_use, text], with:
   124→   124→    // `tool_use` ids were found without `tool_result` blocks immediately after...
   125→   125→    //
   126→   126→    // Reorder that invalid shape into [text] + [tool_use, tool_use]. Consecutive
   127→   127→    // assistant messages are later merged by the provider/SDK, so preserving the
   128→   128→    // original [tool_use...] then [text] order still produces the invalid payload.
   129→   129→    //
   130→   130→    // The root cause appears to be somewhere upstream where the stream is originally
   131→   131→    // processed. We were unable to locate an exact narrower reproduction elsewhere,
   132→   132→    // so we keep this transform in place for the time being.
   133→   133→    msgs = msgs.flatMap((msg) => {
   134→   134→      if (msg.role !== "assistant" || !Array.isArray(msg.content)) return [msg]
   135→   135→
   136→   136→      const parts = msg.content
   137→   137→      const first = parts.findIndex((part) => part.type === "tool-call")
   138→   138→      if (first === -1) return [msg]
   139→   139→      if (!parts.slice(first).some((part) => part.type !== "tool-call")) return [msg]
   140→   140→      return [
   141→   141→        { ...msg, content: parts.filter((part) => part.type !== "tool-call") },
   142→   142→        { ...msg, content: parts.filter((part) => part.type === "tool-call") },
   143→   143→      ]
   144→   144→    })
   145→   145→  }
   146→   146→  if (
   147→   147→    model.providerID === "mistral" ||
   148→   148→    model.api.id.toLowerCase().includes("mistral") ||
   149→   149→    model.api.id.toLocaleLowerCase().includes("devstral")
   150→   150→  ) {
   151→   151→    const scrub = (id: string) => {
   152→   152→      return id
   153→   153→        .replace(/[^a-zA-Z0-9]/g, "") // Remove non-alphanumeric characters
   154→   154→        .substring(0, 9) // Take first 9 characters
   155→   155→        .padEnd(9, "0") // Pad with zeros if less than 9 characters
   156→   156→    }
   157→   157→    const result: ModelMessage[] = []
   158→   158→    for (let i = 0; i < msgs.length; i++) {
   159→   159→      const msg = msgs[i]
   160→   160→      const nextMsg = msgs[i + 1]
   161→   161→
   162→   162→      if (msg.role === "assistant" && Array.isArray(msg.content)) {
   163→   163→        msg.content = msg.content.map((part) => {
   164→   164→          if (part.type === "tool-call" || part.type === "tool-result") {
   165→   165→            return { ...part, toolCallId: scrub(part.toolCallId) }
   166→   166→          }
   167→   167→          return part
   168→   168→        })
   169→   169→      }
   170→   170→      if (msg.role === "tool" && Array.isArray(msg.content)) {
   171→   171→        msg.content = msg.content.map((part) => {
   172→   172→          if (part.type === "tool-result") {
   173→   173→            return { ...part, toolCallId: scrub(part.toolCallId) }
   174→   174→          }
   175→   175→          return part
   176→   176→        })
   177→   177→      }
   178→   178→      result.push(msg)
   179→   179→
   180→   180→      // Fix message sequence: tool messages cannot be followed by user messages
   181→   181→      if (msg.role === "tool" && nextMsg?.role === "user") {
   182→   182→        result.push({
   183→   183→          role: "assistant",
   184→   184→          content: [
   185→   185→            {
   186→   186→              type: "text",
   187→   187→              text: "Done.",
   188→   188→            },
   189→   189→          ],
   190→   190→        })
   191→   191→      }
   192→   192→    }
   193→   193→    return result
   194→   194→  }
   195→   195→
   196→   196→  if (typeof model.capabilities.interleaved === "object" && model.capabilities.interleaved.field) {
   197→   197→    const field = model.capabilities.interleaved.field
   198→   198→    return msgs.map((msg) => {
   199→   199→      if (msg.role === "assistant" && Array.isArray(msg.content)) {
   200→   200→        const reasoningParts = msg.content.filter((part: any) => part.type === "reasoning")
   201→   201→        const reasoningText = reasoningParts.map((part: any) => part.text).join("")
   202→   202→
   203→   203→        // Filter out reasoning parts from content
   204→   204→        const filteredContent = msg.content.filter((part: any) => part.type !== "reasoning")
   205→   205→
   206→   206→        // Include reasoning_content | reasoning_details directly on the message for all assistant messages
   207→   207→        if (reasoningText) {
   208→   208→          return {
   209→   209→            ...msg,
   210→   210→            content: filteredContent,
   211→   211→            providerOptions: {
   212→   212→              ...msg.providerOptions,
   213→   213→              openaiCompatible: {
   214→   214→                ...msg.providerOptions?.openaiCompatible,
   215→   215→                [field]: reasoningText,
   216→   216→              },
   217→   217→            },
   218→   218→          }
   219→   219→        }
   220→   220→
   221→   221→        return {
   222→   222→          ...msg,
   223→   223→          content: filteredContent,
   224→   224→        }
   225→   225→      }
   226→   226→
   227→   227→      return msg
   228→   228→    })
   229→   229→  }
   230→   230→
   231→   231→  return msgs
   232→   232→}

   234→   234→// Determines whether a model's provider respects inline cache_control / cachePoint
   235→   235→// markers. Pure name matching (model.api.id.includes("claude")) is fragile — a Claude
   236→   236→// model behind an OpenAI-compatible proxy gets matched but markers are silently dropped.
   237→   237→// See upstream opencode#26786.
   238→   238→function supportsCacheMarkers(model: Provider.Model): boolean {
   239→   239→  // Anthropic-only SDKs — always support inline markers
   240→   240→  if (model.api.npm === "@ai-sdk/anthropic" || model.api.npm === "@ai-sdk/google-vertex/anthropic") return true
   241→   241→  if (model.providerID === "anthropic" || model.providerID === "google-vertex-anthropic") return true
   242→   242→  // Bedrock cachePoint is a Converse API feature, works across model families
   243→   243→  if (model.api.npm === "@ai-sdk/amazon-bedrock") return true
   244→   244→  // Multi-model providers: only Anthropic/Claude models support cache markers
   245→   245→  if (
   246→   246→    model.api.npm === "@openrouter/ai-sdk-provider" ||
   247→   247→    model.api.npm === "@ai-sdk/github-copilot" ||
   248→   248→    model.api.npm === "@ai-sdk/alibaba"
   249→   249→  ) {
   250→   250→    return (
   251→   251→      model.api.id.includes("claude") ||
   252→   252→      model.api.id.includes("anthropic") ||
   253→   253→      model.id.includes("claude") ||
   254→   254→      model.id.includes("anthropic")
   255→   255→    )
   256→   256→  }
   257→   257→  return false
   258→   258→}

   260→   260→function applyCaching(msgs: ModelMessage[], model: Provider.Model): ModelMessage[] {
   261→   261→  // Only Anthropic and OpenRouter expose a cache-control TTL in their AI SDK;
   262→   262→  // the other providers ignore an unknown `ttl` field, so we only thread it
   263→   263→  // into those two branches. Default (unset) stays the provider 5m default.
   264→   264→  const ttl = model.cachePromptTTL === "1h" ? { ttl: "1h" as const } : {}
   265→   265→  const providerOptions = {
   266→   266→    anthropic: {
   267→   267→      cacheControl: { type: "ephemeral", ...ttl },
   268→   268→    },
   269→   269→    openrouter: {
   270→   270→      cacheControl: { type: "ephemeral", ...ttl },
   271→   271→    },
   272→   272→    bedrock: {
   273→   273→      cachePoint: { type: "default" },
   274→   274→    },
   275→   275→    openaiCompatible: {
   276→   276→      cache_control: { type: "ephemeral" },
   277→   277→    },
   278→   278→    copilot: {
   279→   279→      copilot_cache_control: { type: "ephemeral" },
   280→   280→    },
   281→   281→    alibaba: {
   282→   282→      cacheControl: { type: "ephemeral" },
   283→   283→    },
   284→   284→  }
   285→   285→
   286→   286→  // Strategy: place cache breakpoints at stable prefix boundaries (max 4 allowed by Anthropic)
   287→   287→  // 1. Last system message — system prompt never changes
   288→   288→  // 2. Midpoint of conversation history — long prefix second-level cache
   289→   289→  // 3. Message before the last user message — stable history boundary
   290→   290→  const targets: ModelMessage[] = []
   291→   291→
   292→   292→  const systemMsgs = msgs.filter((msg) => msg.role === "system")
   293→   293→  if (systemMsgs.length > 0) targets.push(systemMsgs[systemMsgs.length - 1])
   294→   294→
   295→   295→  const nonSystem = msgs.filter((msg) => msg.role !== "system")
   296→   296→  const lastUserIdx = nonSystem.findLastIndex((msg) => msg.role === "user")
   297→   297→  if (lastUserIdx >= 1) {
   298→   298→    targets.push(nonSystem[lastUserIdx - 1])
   299→   299→    const midpoint = Math.floor(lastUserIdx / 2)
   300→   300→    if (midpoint > 0 && midpoint < lastUserIdx - 1) targets.push(nonSystem[midpoint])
   301→   301→  } else if (lastUserIdx === 0) {
   302→   302→    targets.push(nonSystem[0])
   303→   303→  }
   304→   304→
   305→   305→  for (const msg of unique(targets)) {
   306→   306→    const useMessageLevelOptions =
   307→   307→      model.providerID === "anthropic" ||
   308→   308→      model.providerID.includes("bedrock") ||
   309→   309→      model.api.npm === "@ai-sdk/amazon-bedrock"
   310→   310→    const shouldUseContentOptions = !useMessageLevelOptions && Array.isArray(msg.content) && msg.content.length > 0
   311→   311→
   312→   312→    if (shouldUseContentOptions) {
   313→   313→      const lastContent = msg.content[msg.content.length - 1]
   314→   314→      if (
   315→   315→        lastContent &&
   316→   316→        typeof lastContent === "object" &&
   317→   317→        lastContent.type !== "tool-approval-request" &&
   318→   318→        lastContent.type !== "tool-approval-response"
   319→   319→      ) {
   320→   320→        lastContent.providerOptions = mergeDeep(lastContent.providerOptions ?? {}, providerOptions)
   321→   321→        continue
   322→   322→      }
   323→   323→    }
   324→   324→
   325→   325→    msg.providerOptions = mergeDeep(msg.providerOptions ?? {}, providerOptions)
   326→   326→  }
   327→   327→
   328→   328→  return msgs
   329→   329→}

   331→   331→function unsupportedParts(msgs: ModelMessage[], model: Provider.Model): ModelMessage[] {
   332→   332→  return msgs.map((msg) => {
   333→   333→    if (msg.role !== "user" || !Array.isArray(msg.content)) return msg
   334→   334→
   335→   335→    const filtered = msg.content.map((part) => {
   336→   336→      if (part.type !== "file" && part.type !== "image") return part
   337→   337→
   338→   338→      // Check for empty base64 image data
   339→   339→      if (part.type === "image") {
   340→   340→        const imageStr = String(part.image)
   341→   341→        if (imageStr.startsWith("data:")) {
   342→   342→          const match = imageStr.match(/^data:([^;]+);base64,(.*)$/)
   343→   343→          if (match && (!match[2] || match[2].length === 0)) {
   344→   344→            return {
   345→   345→              type: "text" as const,
   346→   346→              text: "ERROR: Image file is empty or corrupted. Please provide a valid image.",
   347→   347→            }
   348→   348→          }
   349→   349→        }
   350→   350→      }
   351→   351→
   352→   352→      const mime = part.type === "image" ? String(part.image).split(";")[0].replace("data:", "") : part.mediaType
   353→   353→      const filename = part.type === "file" ? part.filename : undefined
   354→   354→      const modality = mimeToModality(mime)
   355→   355→      if (!modality) return part
   356→   356→      const supported = modality === "image" ? supportsImageInput(model) : model.capabilities.input[modality]
   357→   357→      if (supported) return part
   358→   358→
   359→   359→      const name = filename ? `"${filename}"` : modality
   360→   360→      return {
   361→   361→        type: "text" as const,
   362→   362→        text: `ERROR: Cannot read ${name} (this model does not support ${modality} input). Inform the user.`,
   363→   363→      }
   364→   364→    })
   365→   365→
   366→   366→    return { ...msg, content: filtered }
   367→   367→  })
   368→   368→}

   370→   370→// Returns the decoded byte size of a base64 data URL, or undefined for inputs
   371→   371→// that aren't data URLs (remote URLs, raw binary) and therefore can't be sized.
   372→   372→function imageByteSize(image: string): number | undefined {
   373→   373→  if (!image.startsWith("data:")) return undefined
   374→   374→  const base64 = image.slice(image.indexOf(",") + 1)
   375→   375→  if (!base64) return 0
   376→   376→  const padding = base64.endsWith("==") ? 2 : base64.endsWith("=") ? 1 : 0
   377→   377→  return Math.floor((base64.length * 3) / 4) - padding
   378→   378→}

   380→   380→function limitImages(msgs: ModelMessage[]): ModelMessage[] {
   381→   381→  const maxImages = Flag.MIMOCODE_MAX_PROMPT_IMAGES
   382→   382→  const maxSize = Flag.MIMOCODE_MAX_PROMPT_IMAGE_SIZE
   383→   383→  if (maxImages === undefined && maxSize === undefined) return msgs
   384→   384→
   385→   385→  const total = msgs.reduce(
   386→   386→    (sum, msg) =>
   387→   387→      msg.role === "user" && Array.isArray(msg.content)
   388→   388→        ? sum + msg.content.filter((part) => part.type === "image").length
   389→   389→        : sum,
   390→   390→    0,
   391→   391→  )
   392→   392→  // Drop the oldest excess images so the most recent ones reach the model.
   393→   393→  let toDrop = maxImages === undefined ? 0 : Math.max(0, total - maxImages)
   394→   394→
   395→   395→  return msgs.map((msg) => {
   396→   396→    if (msg.role !== "user" || !Array.isArray(msg.content)) return msg
   397→   397→    const content = msg.content.map((part) => {
   398→   398→      if (part.type !== "image") return part
   399→   399→      if (toDrop > 0) {
   400→   400→        toDrop--
   401→   401→        return { type: "text" as const, text: `[Image omitted: exceeds the configured limit of ${maxImages} prompt image(s).]` }
   402→   402→      }
   403→   403→      if (maxSize !== undefined) {
   404→   404→        const size = imageByteSize(String(part.image))
   405→   405→        if (size !== undefined && size > maxSize) {
   406→   406→          return { type: "text" as const, text: `[Image omitted: exceeds the configured ${maxSize}-byte prompt image size limit.]` }
   407→   407→        }
   408→   408→      }
   409→   409→      return part
   410→   410→    })
   411→   411→    return { ...msg, content }
   412→   412→  })
   413→   413→}

   415→   415→export function message(msgs: ModelMessage[], model: Provider.Model, options: Record<string, unknown>) {
   416→   416→  msgs = unsupportedParts(msgs, model)
   417→   417→  msgs = limitImages(msgs)
   418→   418→  msgs = normalizeMessages(msgs, model, options)
   419→   419→  if (supportsCacheMarkers(model)) {
   420→   420→    msgs = applyCaching(msgs, model)
   421→   421→  }
   422→   422→
   423→   423→  // Remap providerOptions keys from stored providerID to expected SDK key
   424→   424→  const key = sdkKey(model.api.npm)
   425→   425→  if (key && key !== model.providerID) {
   426→   426→    const remap = (opts: Record<string, any> | undefined) => {
   427→   427→      if (!opts) return opts
   428→   428→      if (!(model.providerID in opts)) return opts
   429→   429→      const result = { ...opts }
   430→   430→      result[key] = result[model.providerID]
   431→   431→      delete result[model.providerID]
   432→   432→      return result
   433→   433→    }
   434→   434→
   435→   435→    msgs = msgs.map((msg) => {
   436→   436→      if (!Array.isArray(msg.content)) return { ...msg, providerOptions: remap(msg.providerOptions) }
   437→   437→      return {
   438→   438→        ...msg,
   439→   439→        providerOptions: remap(msg.providerOptions),
   440→   440→        content: msg.content.map((part) => {
   441→   441→          if (part.type === "tool-approval-request" || part.type === "tool-approval-response") {
   442→   442→            return { ...part }
   443→   443→          }
   444→   444→          return { ...part, providerOptions: remap(part.providerOptions) }
   445→   445→        }),
   446→   446→      } as typeof msg
   447→   447→    })
   448→   448→  }
   449→   449→
   450→   450→  return msgs
   451→   451→}

  1120→  1120→// Flatten a root-level `anyOf` / `oneOf` (typically from `z.discriminatedUnion`)
  1121→  1121→// into a single `type: "object"` schema with all variant properties merged at
  1122→  1122→// the root. The discriminator key becomes an `enum`, and per-variant required
  1123→  1123→// fields are encoded as a textual hint on the discriminator's description.
  1124→  1124→//
  1125→  1125→// OpenAI's function-calling validator rejects oneOf/anyOf/allOf/enum/not at the
  1126→  1126→// top level outright, so this is the only shape that gets through. We retain
  1127→  1127→// `additionalProperties: false` to keep the model from inventing fields, and
  1128→  1128→// rely on zod's runtime parse (still using the original discriminated union)
  1129→  1129→// to enforce per-action required fields strictly.
  1130→  1130→function flattenDiscriminatedUnion(schema: JSONSchema.BaseSchema | JSONSchema7): JSONSchema7 {
  1131→  1131→  const root = schema as Record<string, any>
  1132→  1132→  const variants = (root.anyOf ?? root.oneOf) as Array<Record<string, any>> | undefined
  1133→  1133→  if (!variants?.length) return schema as JSONSchema7
  1134→  1134→
  1135→  1135→  // Find the discriminator: a property whose `const` differs across all variants.
  1136→  1136→  const constByKey = new Map<string, Set<unknown>>()
  1137→  1137→  for (const v of variants) {
  1138→  1138→    if (!v.properties) continue
  1139→  1139→    for (const [key, prop] of Object.entries(v.properties as Record<string, any>)) {
  1140→  1140→      if (prop && typeof prop === "object" && "const" in prop) {
  1141→  1141→        if (!constByKey.has(key)) constByKey.set(key, new Set())
  1142→  1142→        constByKey.get(key)!.add(prop.const)
  1143→  1143→      }
  1144→  1144→    }
  1145→  1145→  }
  1146→  1146→  let discriminator: string | undefined
  1147→  1147→  for (const [key, values] of constByKey) {
  1148→  1148→    if (values.size === variants.length) {
  1149→  1149→      discriminator = key
  1150→  1150→      break
  1151→  1151→    }
  1152→  1152→  }
  1153→  1153→
  1154→  1154→  // Merge non-discriminator properties from every variant. Track which variants
  1155→  1155→  // each property appeared in so the description can tell the model
  1156→  1156→  // "(only when action='X'|'Y')" — flat schemas tempt some models (notably
  1157→  1157→  // gpt-5.5) to fill every property regardless of which action they chose.
  1158→  1158→  const properties: Record<string, any> = {}
  1159→  1159→  const propertyOwners: Record<string, unknown[]> = {}
  1160→  1160→  for (const v of variants) {
  1161→  1161→    if (!v.properties) continue
  1162→  1162→    const variantValue = discriminator
  1163→  1163→      ? (v.properties as Record<string, any>)[discriminator]?.const
  1164→  1164→      : undefined
  1165→  1165→    for (const [key, prop] of Object.entries(v.properties as Record<string, any>)) {
  1166→  1166→      if (key === discriminator) continue
  1167→  1167→      if (!(key in properties)) properties[key] = prop
  1168→  1168→      if (variantValue !== undefined) {
  1169→  1169→        if (!propertyOwners[key]) propertyOwners[key] = []
  1170→  1170→        propertyOwners[key].push(variantValue)
  1171→  1171→      }
  1172→  1172→    }
  1173→  1173→  }
  1174→  1174→  if (discriminator) {
  1175→  1175→    for (const [key, owners] of Object.entries(propertyOwners)) {
  1176→  1176→      if (owners.length === variants.length) continue // present in every variant — no annotation
  1177→  1177→      const tag = `(only when ${discriminator}=${owners.map((o) => JSON.stringify(o)).join("|")})`
  1178→  1178→      const original = (properties[key] as Record<string, any>).description as string | undefined
  1179→  1179→      properties[key] = {
  1180→  1180→        ...properties[key],
  1181→  1181→        description: original ? `${tag} ${original}` : tag,
  1182→  1182→      }
  1183→  1183→    }
  1184→  1184→  }
  1185→  1185→
  1186→  1186→  // Discriminator becomes an enum with per-variant required fields hinted in
  1187→  1187→  // its description. Without this hint the model only sees a flat bag of
  1188→  1188→  // optional fields and forgets what to provide for each action.
  1189→  1189→  if (discriminator) {
  1190→  1190→    const proto = (variants[0].properties as Record<string, any>)[discriminator]
  1191→  1191→    const enumValues = variants.map((v) => (v.properties as Record<string, any>)[discriminator!]?.const)
  1192→  1192→    const baseDescription = (proto?.description as string | undefined) ?? ""
  1193→  1193→    const hints = variants
  1194→  1194→      .map((v) => {
  1195→  1195→        const value = (v.properties as Record<string, any>)[discriminator!]?.const
  1196→  1196→        const required = ((v.required as string[] | undefined) ?? []).filter((r) => r !== discriminator)
  1197→  1197→        return required.length > 0 ? `${value}: requires ${required.join(", ")}` : `${value}: no extra required fields`
  1198→  1198→      })
  1199→  1199→      .join("; ")
  1200→  1200→    properties[discriminator] = {
  1201→  1201→      type: "string",
  1202→  1202→      enum: enumValues,
  1203→  1203→      description: baseDescription ? `${baseDescription}\n\nPer-${discriminator}: ${hints}.` : `Per-${discriminator}: ${hints}.`,
  1204→  1204→    }
  1205→  1205→  }
  1206→  1206→
  1207→  1207→  return {
  1208→  1208→    type: "object",
  1209→  1209→    properties,
  1210→  1210→    required: discriminator ? [discriminator] : [],
  1211→  1211→    additionalProperties: false,
  1212→  1212→  } as JSONSchema7
  1213→  1213→}
