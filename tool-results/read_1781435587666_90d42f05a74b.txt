    54→function wrapSSE(res: Response, ms: number, ctl: AbortController) {
    55→  if (typeof ms !== "number" || ms <= 0) return res
    56→  if (!res.body) return res
    57→  if (!res.headers.get("content-type")?.includes("text/event-stream")) return res
    58→
    59→  const reader = res.body.getReader()
    60→  const body = new ReadableStream<Uint8Array>({
    61→    async pull(ctrl) {
    62→      const part = await new Promise<Awaited<ReturnType<typeof reader.read>>>((resolve, reject) => {
    63→        const id = setTimeout(() => {
    64→          const err = new Error("SSE read timed out")
    65→          ctl.abort(err)
    66→          void reader.cancel(err)
    67→          reject(err)
    68→        }, ms)
    69→
    70→        reader.read().then(
    71→          (part) => {
    72→            clearTimeout(id)
    73→            resolve(part)
    74→          },
    75→          (err) => {
    76→            clearTimeout(id)
    77→            reject(err)
    78→          },
    79→        )
    80→      })
    81→
    82→      if (part.done) {
    83→        ctrl.close()
    84→        return
    85→      }
    86→
    87→      ctrl.enqueue(part.value)
    88→    },
    89→    async cancel(reason) {
    90→      ctl.abort(reason)
    91→      await reader.cancel(reason)
    92→    },
    93→  })
    94→
    95→  return new Response(body, {
    96→    headers: new Headers(res.headers),
    97→    status: res.status,
    98→    statusText: res.statusText,
    99→  })
   100→}

  1039→export function fromModelsDevProvider(provider: ModelsDev.Provider): Info {
  1040→  const models: Record<string, Model> = {}

  1042→    models[key] = fromModelsDevModel(provider, model)
  1043→    for (const [mode, opts] of Object.entries(model.experimental?.modes ?? {})) {
  1044→      const id = `${model.id}-${mode}`
  1045→      const base = fromModelsDevModel(provider, model)

  1050→        cost: opts.cost ? mergeDeep(base.cost, cost(opts.cost)) : base.cost,

  1429→      try {
  1430→        using _ = log.time("getSDK", {
  1431→          providerID: model.providerID,

  1434→        const options = { ...provider.options }

  1437→          delete options.fetch

  1439→

  1441→          options["includeUsage"] = true

  1445→          let url =
  1446→            typeof options["baseURL"] === "string" && options["baseURL"] !== "" ? options["baseURL"] : model.api.url
  1447→          if (!url) return

  1475→            providerID: model.providerID,

  1519→

  1521→            ...opts,

  1533→            providerID: model.providerID,

  1568→        throw new InitError({ providerID: model.providerID }, { cause: e })

  1570→    }

  1614→          if (e instanceof NoSuchModelError)
  1615→            throw new ModelNotFoundError(
  1616→              {
  1617→                modelID: model.id,
  1618→                providerID: model.providerID,
  1619→              },
  1620→              { cause: e },
  1621→            )
  1622→          throw e
  1623→        }
  1624→      })
  1625→    })
  1626→
  1627→    const closest = Effect.fn("Provider.closest")(function* (providerID: ProviderID, query: string[]) {
  1628→      const s = yield* InstanceState.get(state)
  1629→      const provider = s.providers[providerID]
  1630→      if (!provider) return undefined
  1631→      for (const item of query) {
  1632→        for (const modelID of Object.keys(provider.models)) {
  1633→          if (modelID.includes(item)) return { providerID, modelID }
  1634→        }
  1635→      }
  1636→      return undefined
  1637→    })
  1638→
  1639→    const resolveModelRef = Effect.fn("Provider.resolveModelRef")(function* (
  1640→      ref: string,
  1641→      contextProviderID?: ProviderID,
  1642→    ) {
  1643→      // Literal "provider/model" — unchanged path.
  1644→      if (ref.includes("/")) {
  1645→        const parsed = parseModel(ref)
  1646→        return yield* getModel(parsed.providerID, parsed.modelID)
  1647→      }
  1648→
  1649→      // Otherwise it's a group name.
  1650→      const cfg = yield* config.get()
  1651→      const group = cfg.model_groups?.[ref]
  1652→      if (!group) {
  1653→        if (BUILTIN_TIERS.has(ref)) {
  1654→          const fallback = yield* defaultModel()
  1655→          return yield* getModel(fallback.providerID, fallback.modelID)
  1656→        }
  1657→        const names = Object.keys(cfg.model_groups ?? {})
  1658→        const matches = fuzzysort.go(ref, names, { limit: 3, threshold: -10000 })
  1659→        throw new ModelGroupNotFoundError({ group: ref, suggestions: matches.map((m) => m.target) })
  1660→      }
  1661→
  1662→      const def = typeof group === "string" ? group : group.default
  1663→      const members = typeof group === "string" ? [group] : (group.models ?? [group.default])
  1664→
  1665→      // Provider-aware: prefer a member on the caller's provider.
  1666→      if (contextProviderID) {
  1667→        for (const member of members) {
  1668→          const parsed = parseModel(member)
  1669→          if (parsed.providerID !== contextProviderID) continue
  1670→          const resolved = yield* getModel(parsed.providerID, parsed.modelID).pipe(Effect.exit)
  1671→          if (resolved._tag === "Success") return resolved.value
  1672→        }
  1673→      }
  1674→
  1675→      // Fall back to the group default.
  1676→      const parsedDefault = parseModel(def)
  1677→      return yield* getModel(parsedDefault.providerID, parsedDefault.modelID)
  1678→    })
  1679→
  1680→    const getSmallModel = Effect.fn("Provider.getSmallModel")(function* (providerID: ProviderID) {
  1681→      const cfg = yield* config.get()
  1682→
  1683→      // Explicit small_model literal wins (back-compat).
  1684→      if (cfg.small_model) {
  1685→        const parsed = parseModel(cfg.small_model)
  1686→        return yield* getModel(parsed.providerID, parsed.modelID)
  1687→      }
  1688→
  1689→      // Otherwise route through the `lite` tier: configured lite group
  1690→      // (provider-aware) → else the built-in tier fallback to defaultModel().
  1691→      return yield* resolveModelRef("lite", providerID)
  1692→    })
  1693→
  1694→    const defaultModel = Effect.fn("Provider.defaultModel")(function* () {
  1695→      const cfg = yield* config.get()
  1696→      if (cfg.model) return parseModel(cfg.model)
  1697→
  1698→      const s = yield* InstanceState.get(state)
  1699→      const recent = yield* fs.readJson(path.join(Global.Path.state, "model.json")).pipe(
  1700→        Effect.map((x): { providerID: ProviderID; modelID: ModelID }[] => {
  1701→          if (!isRecord(x) || !Array.isArray(x.recent)) return []
  1702→          return x.recent.flatMap((item) => {
  1703→            if (!isRecord(item)) return []
  1704→            if (typeof item.providerID !== "string") return []
  1705→            if (typeof item.modelID !== "string") return []
  1706→            return [{ providerID: ProviderID.make(item.providerID), modelID: ModelID.make(item.modelID) }]
  1707→          })
  1708→        }),
  1709→        Effect.catch(() => Effect.succeed([] as { providerID: ProviderID; modelID: ModelID }[])),
  1710→      )
  1711→      for (const entry of recent) {
  1712→        const provider = s.providers[entry.providerID]
  1713→        if (!provider) continue
  1714→        if (!provider.models[entry.modelID]) continue
  1715→        return { providerID: entry.providerID, modelID: entry.modelID }
  1716→      }
  1717→
  1718→      const mimo = s.providers[ProviderID.make("mimo")]
  1719→      if (mimo?.models[ModelID.make("mimo-auto")]) {
  1720→        return { providerID: mimo.id, modelID: ModelID.make("mimo-auto") }
  1721→      }
  1722→
  1723→      const provider = Object.values(s.providers).find((p) => !cfg.provider || Object.keys(cfg.provider).includes(p.id))
  1724→      if (!provider) throw new Error("no providers found")
  1725→      const [model] = sort(Object.values(provider.models))
  1726→      if (!model) throw new Error("no models found")
  1727→      return {
  1728→        providerID: provider.id,
  1729→        modelID: model.id,
  1730→      }
  1731→    })
  1732→
  1733→    return Service.of({ list, getProvider, getModel, getLanguage, closest, getSmallModel, defaultModel, resolveModelRef })
  1734→  }),

  1747→const priority = ["gpt-5", "claude-sonnet-4", "big-pickle", "gemini-3-pro"]
  1748→export function sort<T extends { id: string }>(models: T[]) {
  1749→  return sortBy(
  1750→    models,
  1751→    [(model) => priority.findIndex((filter) => model.id.includes(filter)), "desc"],
  1752→    [(model) => (model.id.includes("latest") ? 0 : 1), "asc"],
  1753→    [(model) => model.id, "desc"],
  1754→  )
  1755→}
