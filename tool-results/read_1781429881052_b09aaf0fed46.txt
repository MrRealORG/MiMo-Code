    35→    35→function levenshteinActor(a: string, b: string): number {
    36→    36→  const m = a.length, n = b.length
    37→    37→  if (m === 0) return n
    38→    38→  if (n === 0) return m
    39→    39→  const dp = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0))
    40→    40→  for (let i = 0; i <= m; i++) dp[i][0] = i
    41→    41→  for (let j = 0; j <= n; j++) dp[0][j] = j
    42→    42→  for (let i = 1; i <= m; i++) {
    43→    43→    for (let j = 1; j <= n; j++) {
    44→    44→      const cost = a[i - 1] === b[j - 1] ? 0 : 1
    45→    45→      dp[i][j] = Math.min(dp[i - 1][j] + 1, dp[i][j - 1] + 1, dp[i - 1][j - 1] + cost)
    46→    46→    }
    47→    47→  }
    48→    48→  return dp[m][n]
    49→    49→}

    54→    54→  return candidates[0].v

    76→    76→// Generic `--name value` / `--name=value` extractor for a fixed set of optional
    77→    77→// flags. Positionals (and any unrecognized tokens) fall through to `rest`.
    78→    78→function extractNamedFlags(
    79→    79→  args: string[],
    80→    80→  names: string[],
    81→    81→  line: number,
    82→    82→): Effect.Effect<{ flags: Record<string, string>; rest: string[] }, { kind: "flag"; line: number; detail: string }> {
    83→    83→  const rest: string[] = []
    84→    84→  const flags: Record<string, string> = {}
    85→    85→  for (let i = 0; i < args.length; i++) {
    86→    86→    const a = args[i]
    87→    87→    const bare = names.find((n) => a === `--${n}`)
    88→    88→    if (bare) {
    89→    89→      const next = args[i + 1]
    90→    90→      if (next === undefined)
    91→    91→        return Effect.fail({ kind: "flag" as const, line, detail: `actor: --${bare} requires a value` })
    92→    92→      flags[bare] = next
    93→    93→      i++
    94→    94→      continue
    95→    95→    }
    96→    96→    const eq = names.find((n) => a.startsWith(`--${n}=`))
    97→    97→    if (eq) {
    98→    98→      const v = a.slice(`--${eq}=`.length)
    99→    99→      if (v === "") return Effect.fail({ kind: "flag" as const, line, detail: `actor: --${eq} requires a value` })
   100→   100→      flags[eq] = v
   101→   101→      continue
   102→   102→    }
   103→   103→    rest.push(a)
   104→   104→  }
   105→   105→  return Effect.succeed({ flags, rest })
   106→   106→}

   199→   199→export function parseActorScript(
   200→   200→  script: string,
   201→   201→): Effect.Effect<ActorShellArgs[], unknown> {
   202→   202→  return Effect.gen(function* () {
   203→   203→    const argvList = yield* tokenize(script)
   204→   204→    const out: ActorShellArgs[] = []
   205→   205→    for (const argv of argvList) {
   206→   206→      const [head, verb, ...rest] = argv.tokens
   207→   207→      if (head !== "actor") {
   208→   208→        return yield* Effect.fail({
   209→   209→          kind: "unknown-verb",
   210→   210→          line: argv.line,
   211→   211→          detail: `actor: every command must start with 'actor' (got '${head ?? ""}')`,
   212→   212→        })
   213→   213→      }
   214→   214→      const parsed = yield* mapActorVerb(verb, rest, argv.line)
   215→   215→      out.push(parsed)
   216→   216→    }
   217→   217→    return out
   218→   218→  })
   219→   219→}

   221→   221→function inferAction(o: Record<string, unknown>): "run" | "spawn" {
   222→   222→  if (o.action === "spawn" || o.action === "run") return o.action
   223→   223→  if (o.background === true || o.async === true) return "spawn"
   224→   224→  return "run"
   225→   225→}

   227→   227→// Recover a shell-mode actor call that arrived shaped like the JSON tool args
   228→   228→// (no `script`): the Task-prior bare `{subagent_type, description, prompt}`, a
   229→   229→// stringified `{operation:"..."}` envelope, or an already-nested `{operation:{}}`.
   230→   230→// Returns the parsed shape for shellWrap to route to execute (which zod-validates
   231→   231→// it), or undefined if rawArgs can't be lifted.
   232→   232→export function recoverActorArgs(rawArgs: unknown): ActorShellArgs | undefined {
   233→   233→  if (rawArgs == null || typeof rawArgs !== "object") return undefined
   234→   234→  let obj = rawArgs as Record<string, unknown>
   235→   235→  if (typeof obj.operation === "string") {
   236→   236→    try {
   237→   237→      const inner = JSON.parse(obj.operation)
   238→   238→      if (inner && typeof inner === "object" && !Array.isArray(inner)) obj = { operation: inner }
   239→   239→    } catch {}
   240→   240→  }
   241→   241→  if (obj.operation && typeof obj.operation === "object" && !Array.isArray(obj.operation))
   242→   242→    return { operation: obj.operation } as ActorShellArgs
   243→   243→  const subagent_type = obj.subagent_type
   244→   244→  const description = obj.description
   245→   245→  const prompt = obj.prompt
   246→   246→  if (typeof subagent_type === "string" && typeof description === "string" && typeof prompt === "string") {
   247→   247→    const op: Record<string, unknown> = { action: inferAction(obj), subagent_type, description, prompt }
   248→   248→    // Carry only the optional fields a confused model plausibly puts at top level
   249→   249→    // alongside the bare Task-prior triple. This is a deliberate subset of the
   250→   250→    // run/spawn schema's optionals (model, actor_id, timeout_ms, command, context,
   251→   251→    // task_id, output_schema) — the others (timeout_ms/command/context/output_schema)
   252→   252→    // are dropped here, falling back to their schema defaults. Low risk in practice:
   253→   253→    // the bare shape mimo emits is the 3 required fields, rarely with extras. When
   254→   254→    // adding an actor schema field, decide whether bare-shape recover should carry
   255→   255→    // it here, or this whitelist silently drifts from the schema.
   256→   256→    if (typeof obj.model === "string") op.model = obj.model
   257→   257→    if (typeof obj.task_id === "string") op.task_id = obj.task_id
   258→   258→    if (typeof obj.actor_id === "string") op.actor_id = obj.actor_id
   259→   259→    return { operation: op } as ActorShellArgs
   260→   260→  }
   261→   261→  return undefined
   262→   262→}

   396→   396→            "(optional) If this subagent is doing work for a specific task in the `task` tool, pass that task's ID (e.g. T4, T2.1) here — only an ID the `task` tool returned this session. After completion, the actor.postStop hook validates that tasks/<task_id>/progress.md exists with the required sections. If the ID is malformed or names no existing task, the binding is silently dropped and the subagent's findings are NOT captured to that task. Leave omitted only for work that isn't tied to a task.",
   397→   397→          ),
   398→   398→        output_schema: z
   399→   399→          .record(z.string(), z.any())
   400→   400→          .optional()
   401→   401→          .describe(
   402→   402→            "(optional) A JSON Schema. When set, the subagent is forced to return a single structured object matching this schema (via the StructuredOutput tool) instead of free text.",
   403→   403→          ),
   404→   404→      })
   405→   405→
   406→   406→      const statusSchema = z.strictObject({
   407→   407→        action: z.literal("status"),
   408→   408→        actor_id: actorIdRequiredField,
   409→   409→      })
   410→   410→
   411→   411→      const waitSchema = z.strictObject({
   412→   412→        action: z.literal("wait"),
   413→   413→        actor_id: actorIdRequiredField,
   414→   414→        timeout_ms: timeoutField,
   415→   415→      })
   416→   416→
   417→   417→      const cancelSchema = z.strictObject({
   418→   418→        action: z.literal("cancel"),
   419→   419→        actor_id: actorIdRequiredField,
   420→   420→      })
   421→   421→
   422→   422→      const sendSchema = z.strictObject({
   423→   423→        action: z.literal("send"),
   424→   424→        to_session_id: z
   425→   425→          .string()
   426→   426→          .min(1)
   427→   427→          .optional()
   428→   428→          .describe(
   429→   429→            "(optional) Target session ID. Defaults to the current session — useful for sending to subagents in this session.",
   430→   430→          ),
   431→   431→        to_actor_id: z
   432→   432→          .string()
   433→   433→          .min(1)
   434→   434→          .describe(
   435→   435→            "Target actor ID. Use 'main' to send to a session's main agent, or a subagent ID like 'explore-1'.",
   436→   436→          ),
   437→   437→        content: z.string().min(1).describe("Message content (plain text). Wrapped in <inbox> for the receiver."),
   438→   438→        type: z
   439→   439→          .string()
   440→   440→          .optional()
   441→   441→          .describe(
   442→   442→            "(optional) Message type. Default 'text' is wrapped in <inbox>...</inbox>. 'actor_notification' is passed through verbatim (sender pre-renders).",
   443→   443→          ),
   444→   444→      })
   445→   445→
   446→   446→      const parameters = z.strictObject({
   447→   447→        // .meta({ type: "object" }) is REQUIRED — without it the emitted JSON
   448→   448→        // schema's `operation` node has only `anyOf`, no `type`, and some models
   449→   449→        // (notably mimo-v2.5-pro) stringify the whole envelope
   450→   450→        // ({"operation":"{\"action\":\"run\",...}"}) which fails zod validation.
   451→   451→        // The root strictObject also means flattenDiscriminatedUnion finds no
   452→   452→        // root-level union and passes through unchanged — root keeps exactly one
   453→   453→        // key (`operation`), so models can't drop the discriminator.
   454→   454→        operation: z
   455→   455→          .discriminatedUnion("action", [
   456→   456→            runSchema,
   457→   457→            spawnSchema,
   458→   458→            statusSchema,
   459→   459→            waitSchema,
   460→   460→            cancelSchema,
   461→   461→            sendSchema,
   462→   462→          ])
   463→   463→          .meta({ type: "object" }),
   464→   464→      })
   465→   465→
   466→   466→      const run = Effect.fn("ActorTool.execute")(function* (input: z.infer<typeof parameters>, ctx: Tool.Context) {
   467→   467→        const op = input.operation
   468→   468→        const cfg = yield* config.get()
   469→   469→
   470→   470→        // Helper: "actor belongs to another session OR doesn't exist" response.
   471→   471→        // Same response for both cases — don't leak the difference (POSIX: you can
   472→   472→        // only reap your own children).
   473→   473→        const unknownResponse = (label: string, actorID: string) => {
   474→   474→          const snapshot = { status: "unknown" as const, actor_id: actorID }
   475→   475→          return {
   476→   476→            title: `Actor ${label}: unknown`,
   477→   477→            output: JSON.stringify(snapshot),
   478→   478→            metadata: { actor_id: actorID, status: "unknown" } as Record<string, any>,
   479→   479→          }
   480→   480→        }
   481→   481→
   482→   482→        // Look up an actor in the registry. Subagent actors live under
   483→   483→        // ctx.sessionID; peer actors live under their own sessionID (== actorID).
   484→   484→        // Try the subagent location first, then fall back to the peer location.
   485→   485→        const findActor = Effect.fn("ActorTool.findActor")(function* (actorID: string) {
   486→   486→          const sub = yield* actorRegistry.get(ctx.sessionID, actorID)
   487→   487→          if (sub) return { entry: sub, sessionID: ctx.sessionID }
   488→   488→          const sid = SessionID.make(actorID)
   489→   489→          const peer = yield* actorRegistry.get(sid, actorID)
   490→   490→          if (peer) return { entry: peer, sessionID: sid }
   491→   491→          return undefined
   492→   492→        })
   493→   493→
   494→   494→        if (op.action ==="send") {
   495→   495→          const inboxSvc = inboxServiceRef.current
   496→   496→          if (!inboxSvc) {
   497→   497→            return yield* Effect.fail(
   498→   498→              new Error("Inbox service unavailable — Inbox.layer must be running for the actor tool to send messages"),
   499→   499→            )
   500→   500→          }
   501→   501→          const targetSid = op.to_session_id !== undefined ? SessionID.make(op.to_session_id) : ctx.sessionID
   502→   502→          const sendResult = yield* inboxSvc
   503→   503→            .send({
   504→   504→              receiverSessionID: targetSid,
   505→   505→              receiverActorID: op.to_actor_id,
   506→   506→              senderSessionID: ctx.sessionID,
   507→   507→              senderActorID: ctx.agent ?? "main",
   508→   508→              content: op.content,
   509→   509→              ...(op.type !== undefined ? { type: op.type } : {}),
   510→   510→            })
   511→   511→            .pipe(
   512→   512→              Effect.catchTag("InboxReceiverNotFound", () =>
   513→   513→                Effect.succeed({ inboxID: null as string | null, error: "receiver not found" }),
   514→   514→              ),
   515→   515→            )
   516→   516→          if ("error" in sendResult) {
   517→   517→            return {
   518→   518→              title: `Send failed: receiver not found`,
   519→   519→              output: JSON.stringify(sendResult),
   520→   520→              metadata: {
   521→   521→                receiver_actor_id: op.to_actor_id,
   522→   522→                receiver_session_id: targetSid,
   523→   523→                error: sendResult.error,
   524→   524→              } as Record<string, any>,
   525→   525→            }
   526→   526→          }
   527→   527→          return {
   528→   528→            title: `Sent to ${op.to_actor_id}`,
   529→   529→            output: JSON.stringify({ inboxID: sendResult.inboxID }),
   530→   530→            metadata: {
   531→   531→              inboxID: sendResult.inboxID,
   532→   532→              receiver_actor_id: op.to_actor_id,
   533→   533→              receiver_session_id: targetSid,
   534→   534→            } as Record<string, any>,
   535→   535→          }
   536→   536→        }
   537→   537→
   538→   538→        if (op.action ==="status") {
   539→   539→          const found = yield* findActor(op.actor_id)
   540→   540→          if (!found) return unknownResponse("status", op.actor_id)
   541→   541→          const entry = found.entry
   542→   542→          const snapshot = {
   543→   543→            status: entry.status,
   544→   544→            actor_id: entry.actorID,
   545→   545→            description: entry.description,
   546→   546→            agent: entry.agent,
   547→   547→            background: entry.background,
   548→   548→            turnCount: entry.turnCount,
   549→   549→            lastTurnTime: entry.lastTurnTime,
   550→   550→            ...(entry.lastError !== undefined ? { error: entry.lastError } : {}),
   551→   551→            time: entry.time,
   552→   552→          }
   553→   553→          return {
   554→   554→            title: `Actor status: ${entry.status}`,
   555→   555→            output: JSON.stringify(snapshot),
   556→   556→            metadata: { actor_id: entry.actorID, status: entry.status } as Record<string, any>,
   557→   557→          }
   558→   558→        }
   559→   559→
   560→   560→        if (op.action ==="wait") {
   561→   561→          const found = yield* findActor(op.actor_id)
   562→   562→          if (!found) return unknownResponse("wait", op.actor_id)
   563→   563→          const snap = yield* waiter.wait({
   564→   564→            sessionID: found.sessionID,
   565→   565→            actor_id: op.actor_id,
   566→   566→            timeout_ms: op.timeout_ms,
   567→   567→          })
   568→   568→          return {
   569→   569→            title: `Actor wait: ${snap.status}${snap.lastOutcome ? "/" + snap.lastOutcome : ""}`,
   570→   570→            output: JSON.stringify(snap),
   571→   571→            metadata: {
   572→   572→              actor_id: snap.actor_id,
   573→   573→              status: snap.status,
   574→   574→              ...(snap.lastOutcome ? { lastOutcome: snap.lastOutcome } : {}),
   575→   575→            } as Record<string, any>,
   576→   576→          }
   577→   577→        }
   578→   578→
   579→   579→        if (op.action ==="cancel") {
   580→   580→          const found = yield* findActor(op.actor_id)
   581→   581→          if (!found) return unknownResponse("cancel", op.actor_id)
   582→   582→          const entry = found.entry
   583→   583→
   584→   584→          // Already terminal? No-op — return current status. Idempotent.
   585→   585→          if (entry.status === "idle") {
   586→   586→            const snapshot = {
   587→   587→              status: entry.status,
   588→   588→              actor_id: entry.actorID,
   589→   589→              description: entry.description,
   590→   590→              agent: entry.agent,
   591→   591→              background: entry.background,
   592→   592→            }
   593→   593→            return {
   594→   594→              title: `Actor cancel: ${entry.status}`,
   595→   595→              output: JSON.stringify(snapshot),
   596→   596→              metadata: { actor_id: entry.actorID, status: entry.status } as Record<string, any>,
   597→   597→            }
   598→   598→          }
   599→   599→
   600→   600→          // Signal the actor through Actor.cancel — marks status "cancelled" in the registry.
   601→   601→          const actorForCancel = yield* requireActor()
   602→   602→          yield* actorForCancel.cancel(found.sessionID, entry.actorID, "graceful")
   603→   603→
   604→   604→          const snapshot = {
   605→   605→            status: "cancelled" as const,
   606→   606→            actor_id: entry.actorID,
   607→   607→            description: entry.description,
   608→   608→            agent: entry.agent,
   609→   609→            background: entry.background,
   610→   610→          }
   611→   611→          return {
   612→   612→            title: `Actor cancel: cancelled`,
   613→   613→            output: JSON.stringify(snapshot),
   614→   614→            metadata: { actor_id: entry.actorID, status: "cancelled" } as Record<string, any>,
   615→   615→          }
   616→   616→        }
   617→   617→
   618→   618→        // op.action ==="run" or "spawn" — schema guarantees
   619→   619→        // description / prompt / subagent_type are present and non-empty.
   620→   620→        if (!ctx.extra?.bypassAgentCheck) {
   621→   621→          yield* ctx.ask({
   622→   622→            permission: "actor",
   623→   623→            patterns: [op.subagent_type],
   624→   624→            always: ["*"],
   625→   625→            metadata: {
   626→   626→              description: op.description,
   627→   627→              subagent_type: op.subagent_type,
   628→   628→            },
   629→   629→          })
   630→   630→        }
   631→   631→
   632→   632→        const next = yield* agent.get(op.subagent_type)
   633→   633→        if (!next) {
   634→   634→          return yield* Effect.fail(new Error(`Unknown agent type: ${op.subagent_type} is not a valid agent type`))
   635→   635→        }
   636→   636→
   637→   637→        let prompt = op.prompt
   638→   638→        const background = op.action ==="spawn"
   639→   639→
   640→   640→        // Inject checkpoint summaries for context="state" mode
   641→   641→        if (op.context === "state") {
   642→   642→          const latest = yield* checkpoint
   643→   643→            .loadLatest(ctx.sessionID)
   644→   644→            .pipe(Effect.catch(() => Effect.succeed(undefined)))
   645→   645→          if (latest) {
   646→   646→            prompt =
   647→   647→              [
   648→   648→                "<session-state>",
   649→   649→                "Here is a summary of the parent session's progress:",
   650→   650→                "",
   651→   651→                latest,
   652→   652→                "</session-state>",
   653→   653→                "",
   654→   654→              ].join("\n") + prompt
   655→   655→          }
   656→   656→          // If no checkpoint, fall through — child gets just the prompt (same as "none")
   657→   657→        }
   658→   658→
   659→   659→        const msg = yield* Effect.sync(() => MessageV2.get({ sessionID: ctx.sessionID, messageID: ctx.messageID }))
   660→   660→        if (msg.info.role !== "assistant") return yield* Effect.fail(new Error("Not an assistant message"))
   661→   661→
   662→   662→        const modelRef = op.model ?? next.modelRef
   663→   663→        const model = modelRef
   664→   664→          ? yield* provider
   665→   665→              .resolveModelRef(modelRef, msg.info.providerID)
   666→   666→              .pipe(Effect.map((m) => ({ modelID: m.id, providerID: m.providerID })))
   667→   667→          : (next.model ?? {
   668→   668→              modelID: msg.info.modelID,
   669→   669→              providerID: msg.info.providerID,
   670→   670→            })
   671→   671→
   672→   672→        // Validate task_id by reference at execute time (NOT in the schema, so a
   673→   673→        // bad value degrades instead of hard-failing the call). A malformed shape
   674→   674→        // or an ID that names no task in this session ⇒ run ad-hoc (task_id
   675→   675→        // dropped) and tell the model why, so a fabricated ID becomes harmless
   676→   676→        // instead of triggering phantom postStop progress nagging.
   677→   677→        let effectiveTaskId = op.task_id
   678→   678→        let taskNotice = ""
   679→   679→        if (op.task_id) {
   680→   680→          if (!TaskID.safeParse(op.task_id).success) {
   681→   681→            effectiveTaskId = undefined
   682→   682→            taskNotice = `note: task_id "${op.task_id}" is not a valid task ID (expected Tn or Tn.m); ran ad-hoc. Task IDs come from the \`task\` tool.`
   683→   683→          } else {
   684→   684→            const existing = yield* taskRegistry.get({ session_id: ctx.sessionID, id: op.task_id })
   685→   685→            if (!existing) {
   686→   686→              effectiveTaskId = undefined
   687→   687→              taskNotice = `note: task_id "${op.task_id}" does not exist in this session; ran ad-hoc. Create it with the \`task\` tool first, or omit task_id.`
   688→   688→            }
   689→   689→          }
   690→   690→        }
   691→   691→
   692→   692→        // v6: subagents share the parent's sessionID and run as registered actors
   693→   693→        // under the parent. Actor.spawn handles registry registration, forking
   694→   694→        // the agent loop, and sending inbox notifications on terminal — replacing
   695→   695→        // the legacy session.create + manual fork path that lived here pre-Task-29.
   696→   696→        const actor = yield* requireActor()
   697→   697→        const spawnResult = yield* actor.spawn({
   698→   698→          mode: "subagent",
   699→   699→          sessionID: ctx.sessionID,
   700→   700→          agentType: next.name,
   701→   701→          description: op.description,
   702→   702→          task: prompt,
   703→   703→          context: op.context ?? "none",
   704→   704→          tools: next.toolAllowlist ? [...next.toolAllowlist] : "INHERIT",
   705→   705→          model,
   706→   706→          background,
   707→   707→          task_id: effectiveTaskId,
   708→   708→          ...(op.output_schema
   709→   709→            ? { format: { type: "json_schema" as const, schema: op.output_schema, retryCount: 2 } }
   710→   710→            : {}),
   711→   711→        })
   712→   712→
   713→   713→        yield* ctx.metadata({
   714→   714→          title: op.description,
   715→   715→          metadata: {
   716→   716→            sessionId: spawnResult.sessionID,
   717→   717→            actorId: spawnResult.actorID,
   718→   718→            model,
   719→   719→          },
   720→   720→        })
   721→   721→
   722→   722→        if (op.action ==="spawn") {
   723→   723→          return {
   724→   724→            title: op.description,
   725→   725→            metadata: { sessionId: spawnResult.sessionID, actorId: spawnResult.actorID, model },
   726→   726→            output:
   727→   727→              (taskNotice ? taskNotice + "\n" : "") +
   728→   728→              `Background actor started. actor_id: ${spawnResult.actorID}\nThe result will be delivered as a notification when complete.`,
   729→   729→          }
   730→   730→        }
   731→   731→
   732→   732→        // op.action ==="run": blocking path — await the authoritative
   733→   733→        // `outcome` Deferred. It is resolved in spawn's onSuccess AFTER the
   734→   734→        // preStop loop AND the completion gate (but before the fire-and-forget
   735→   735→        // postStop loop), so the parent sees the reconciled status/summary —
   736→   736→        // unlike ActorWaiter, which resolves on the row's first `idle` and would
   737→   737→        // miss the gate's downgrade.
   738→   738→        function cancelHandler() {
   739→   739→          Effect.runFork(actor.cancel(spawnResult.sessionID, spawnResult.actorID, "graceful"))
   740→   740→        }
   741→   741→        const outcome = yield* Effect.acquireUseRelease(
   742→   742→          Effect.sync(() => {
   743→   743→            ctx.abort.addEventListener("abort", cancelHandler)
   744→   744→          }),
   745→   745→          () =>
   746→   746→            Deferred.await(spawnResult.outcome).pipe(
   747→   747→              Effect.timeout(op.timeout_ms ?? 600_000),
   748→   748→              Effect.catchTag("TimeoutError", () => Effect.succeed({ status: "timeout" as const })),
   749→   749→            ),
   750→   750→          () =>
   751→   751→            Effect.sync(() => {
   752→   752→              ctx.abort.removeEventListener("abort", cancelHandler)
   753→   753→            }),
   754→   754→        )
   755→   755→
   756→   756→        // Blocking run preserves the pre-unification contract: tool call fails
   757→   757→        // when the child fails. The LLM sees a tool error, not a "success with
   758→   758→        // error in output." (The explicit action="wait" returns the structured
   759→   759→        // snapshot as a regular tool result — that's a different contract.)
   760→   760→        if (outcome.status === "failure") {
   761→   761→          return yield* Effect.fail(new Error(`Tool execution failed: ${outcome.error ?? "unknown"}`))
   762→   762→        }
   763→   763→
   764→   764→        const resultText =
   765→   765→          outcome.status === "success"
   766→   766→            ? outcome.structured !== undefined
   767→   767→              ? JSON.stringify(outcome.structured)
   768→   768→              : (outcome.finalText ?? "(no output)")
   769→   769→            : outcome.status === "timeout"
   770→   770→              ? "<timeout>task did not complete within timeout</timeout>"
   771→   771→              : "<cancelled>task was cancelled</cancelled>"
   772→   772→        const statusAttr = outcome.status === "success" ? (outcome.reportedStatus ?? "unknown") : outcome.status
   773→   773→        const summaryAttr =
   774→   774→          outcome.status === "success" && outcome.reportedSummary
   775→   775→            ? ` summary="${outcome.reportedSummary.replace(/\s+/g, " ").replace(/"/g, "'").trim()}"`
   776→   776→            : ""
   777→   777→        return {
   778→   778→          title: op.description,
   779→   779→          metadata: { sessionId: spawnResult.sessionID, actorId: spawnResult.actorID, model } as Record<string, any>,
   780→   780→          output: [
   781→   781→            ...(taskNotice ? [taskNotice, ""] : []),
   782→   782→            `actor_id: ${spawnResult.actorID} (for resuming to continue this task if needed)`,
   783→   783→            "",
   784→   784→            `<actor_result status="${statusAttr}"${summaryAttr}>`,
   785→   785→            resultText,
   786→   786→            "</actor_result>",
   787→   787→          ].join("\n"),
   788→   788→        }
   789→   789→      })
   790→   790→
   791→   791→      return {
   792→   792→        description: DESCRIPTION,
   793→   793→        parameters,
   794→   794→        execute: (input: z.infer<typeof parameters>, ctx: Tool.Context) => run(input, ctx).pipe(Effect.orDie),
   795→   795→        shell: {
   796→   796→          description: SHELL_DESCRIPTION,
   797→   797→          parse: parseActorScript,
   798→   798→          recover: recoverActorArgs,
   799→   799→        },
   800→   800→      }
   801→   801→    })
   802→   802→  }),
   803→   803→)
   804→   804→
