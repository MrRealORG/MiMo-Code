    50→export function fromRow(row: SessionRow): Info {
    51→  const summary =
    52→    row.summary_additions !== null || row.summary_deletions !== null || row.summary_files !== null
    53→      ? {
    54→          additions: row.summary_additions ?? 0,
    55→          deletions: row.summary_deletions ?? 0,

    67→    directory: row.directory,

   112→function getForkedTitle(title: string): string {
   113→  const match = title.match(/^(.+) \(fork #(\d+)\)$/)
   114→  if (match) {
   115→    const base = match[1]
   116→    const num = parseInt(match[2], 10)
   117→    return `${base} (fork #${num + 1})`
   118→  }
   119→  return `${title} (fork #1)`
   120→}

   282→export const getUsage = (input: { model: Provider.Model; usage: LanguageModelUsage; metadata?: ProviderMetadata }) => {
   283→  const safe = (value: number) => {
   284→    if (!Number.isFinite(value)) return 0
   285→    return value
   286→  }
   287→  const inputTokens = safe(input.usage.inputTokens ?? 0)
   288→  const outputTokens = safe(input.usage.outputTokens ?? 0)
   289→  const reasoningTokens = safe(input.usage.outputTokenDetails?.reasoningTokens ?? input.usage.reasoningTokens ?? 0)
   290→
   291→  const cacheReadInputTokens = safe(
   292→    input.usage.inputTokenDetails?.cacheReadTokens ?? input.usage.cachedInputTokens ?? 0,
   293→  )
   294→  const cacheWriteInputTokens = safe(
   295→    Number(
   296→      input.usage.inputTokenDetails?.cacheWriteTokens ??
   297→        input.metadata?.["anthropic"]?.["cacheCreationInputTokens"] ??
   298→        // google-vertex-anthropic returns metadata under "vertex" key
   299→        // (AnthropicMessagesLanguageModel custom provider key from 'vertex.anthropic.messages')
   300→        input.metadata?.["vertex"]?.["cacheCreationInputTokens"] ??
   301→        // @ts-expect-error
   302→        input.metadata?.["bedrock"]?.["usage"]?.["cacheWriteInputTokens"] ??
   303→        // @ts-expect-error
   304→        input.metadata?.["venice"]?.["usage"]?.["cacheCreationInputTokens"] ??
   305→        0,
   306→    ),
   307→  )
   308→
   309→  // AI SDK v6 normalized inputTokens to include cached tokens across all providers
   310→  // (including Anthropic/Bedrock which previously excluded them). Always subtract cache
   311→  // tokens to get the non-cached input count for separate cost calculation.
   312→  const adjustedInputTokens = safe(inputTokens - cacheReadInputTokens - cacheWriteInputTokens)
   313→
   314→  const total = input.usage.totalTokens
   315→
   316→  const tokens = {
   317→    total,
   318→    input: adjustedInputTokens,
   319→    output: safe(outputTokens - reasoningTokens),
   320→    reasoning: reasoningTokens,
   321→    cache: {
   322→      write: cacheWriteInputTokens,
   323→      read: cacheReadInputTokens,
   324→    },
   325→  }
   326→
   327→  const costInfo =
   328→    input.model.cost?.experimentalOver200K && tokens.input + tokens.cache.read > 200_000
   329→      ? input.model.cost.experimentalOver200K
   330→      : input.model.cost
   331→  return {
   332→    cost: safe(
   333→      new Decimal(0)
   334→        .add(new Decimal(tokens.input).mul(costInfo?.input ?? 0).div(1_000_000))
   335→        .add(new Decimal(tokens.output).mul(costInfo?.output ?? 0).div(1_000_000))
   336→        .add(new Decimal(tokens.cache.read).mul(costInfo?.cache?.read ?? 0).div(1_000_000))
   337→        .add(new Decimal(tokens.cache.write).mul(costInfo?.cache?.write ?? 0).div(1_000_000))
   338→        // TODO: update models.dev to have better pricing model, for now:
   339→        // charge reasoning tokens at the same rate as output tokens
   340→        .add(new Decimal(tokens.reasoning).mul(costInfo?.output ?? 0).div(1_000_000))
   341→        .toNumber(),
   342→    ),
   343→    tokens,
   344→  }
   345→}

   493→    const get = Effect.fn("Session.get")(function* (id: SessionID) {
   494→      const row = yield* db((d) => d.select().from(SessionTable).where(eq(SessionTable.id, id)).get())

   510→    const remove: Interface["remove"] = Effect.fnUntraced(function* (sessionID: SessionID) {
   511→      try {
   512→        const session = yield* get(sessionID)
   513→        const kids = yield* children(sessionID)
   514→        for (const child of kids) {

   518→        // `remove` needs to work in all cases, such as a broken

   526→
   527→        yield* Effect.sync(() => {
   528→          SyncEvent.run(Event.Deleted, { sessionID, info: session }, { publish: hasInstance })

   530→        })

   566→          .get(),
   567→      )
   568→      if (!row) return
   569→      return {
   570→        ...row.data,
   571→        id: row.id,
   572→        sessionID: row.session_id,
   573→        messageID: row.message_id,
   574→      } as MessageV2.Part
   575→    })

   577→    const create = Effect.fn("Session.create")(function* (input?: {
   578→      parentID?: SessionID
   579→      contextFrom?: SessionID
   580→      contextWatermark?: MessageID
   581→      title?: string
   582→      permission?: Permission.Ruleset
   583→      workspaceID?: WorkspaceID
   584→    }) {
   585→      const directory = yield* InstanceState.directory
   586→      const workspace = yield* InstanceState.workspaceID

   589→        contextFrom: input?.contextFrom,

   595→      })

   598→    const fork = Effect.fn("Session.fork")(function* (input: { sessionID: SessionID; messageID?: MessageID }) {
   599→      const directory = yield* InstanceState.directory
   600→      const original = yield* get(input.sessionID)
   601→      const title = getForkedTitle(original.title)
   602→      const session = yield* createNext({
   603→        directory,
   604→        workspaceID: original.workspaceID,
   605→        title,
   606→      })
   607→      const msgs = yield* messages({ sessionID: input.sessionID, agentID: "*" })
   608→      const idMap = new Map<string, MessageID>()
   609→
   610→      for (const msg of msgs) {
   611→        if (input.messageID && msg.info.id >= input.messageID) break
   612→        const newID = MessageID.ascending()
   613→        idMap.set(msg.info.id, newID)
   614→
   615→        const parentID = msg.info.role === "assistant" && msg.info.parentID ? idMap.get(msg.info.parentID) : undefined
   616→        const cloned = yield* updateMessage({
   617→          ...msg.info,
   618→          sessionID: session.id,
   619→          id: newID,
   620→          ...(parentID && { parentID }),
   621→        })
   622→
   623→        for (const part of msg.parts) {
   624→          yield* updatePart({
   625→            ...part,
   626→            id: PartID.ascending(),
   627→            messageID: cloned.id,
   628→            sessionID: session.id,
   629→          })
   630→        }
   631→      }
   632→      return session
   633→    })

   735→    /** Finds the first message matching the predicate, searching newest-first.
   736→     *
   737→     * Slice contract: `options.agentID` defaults to `"main"` (mirrors `messages()`).
   738→     * Pass `"*"` for cross-slice lookup. */
   739→    const findMessage = Effect.fn("Session.findMessage")(function* (
   740→      sessionID: SessionID,
   741→      predicate: (msg: MessageV2.WithParts) => boolean,
   742→      options?: { agentID?: string },
   743→    ) {
   744→      for (const item of MessageV2.stream(sessionID, { agentID: options?.agentID })) {
   745→        if (predicate(item)) return Option.some(item)
   746→      }
   747→      return Option.none<MessageV2.WithParts>()
   748→    })

   750→    const lastMainMessageID = Effect.fn("Session.lastMainMessageID")(function* (sessionID: SessionID) {

   756→          .orderBy(desc(MessageTable.id))
   757→          .limit(1)
   758→          .get(),
   759→      )
   760→      return row?.id
   761→    })

   796→export function* list(input?: {
   797→  directory?: string
   798→  workspaceID?: WorkspaceID
   799→  roots?: boolean
   800→  start?: number
   801→  search?: string
   802→  limit?: number
   803→}) {
   804→  const project = Instance.project
   805→  const conditions = [eq(SessionTable.project_id, project.id)]
   806→
   807→  if (input?.workspaceID) {
   808→    conditions.push(eq(SessionTable.workspace_id, input.workspaceID))
   809→  }
   810→  if (!Flag.MIMOCODE_EXPERIMENTAL_WORKSPACES) {
   811→    if (input?.directory) {
   812→      conditions.push(eq(SessionTable.directory, input.directory))
   813→    }
   814→  }
   815→  if (input?.roots) {
   816→    conditions.push(isNull(SessionTable.parent_id))
   817→  }
   818→  if (input?.start) {
   819→    conditions.push(gte(SessionTable.time_updated, input.start))
   820→  }
   821→  if (input?.search) {
   822→    conditions.push(like(SessionTable.title, `%${input.search}%`))
   823→  }
   824→
   825→  const limit = input?.limit ?? 100
   826→
   827→  const rows = Database.use((db) =>
   828→    db
   829→      .select()
   830→      .from(SessionTable)
   831→      .where(and(...conditions))
   832→      .orderBy(desc(SessionTable.time_updated))
   833→      .limit(limit)
   834→      .all(),
   835→  )
   836→  for (const row of rows) {
   837→    yield fromRow(row)
   838→  }
   839→}

   841→export function* listGlobal(input?: {
   842→  directory?: string
   843→  roots?: boolean
   844→  start?: number
   845→  cursor?: number
   846→  search?: string
   847→  limit?: number
   848→  archived?: boolean
   849→}) {
   850→  const conditions: SQL[] = []
   851→
   852→  if (input?.directory) {
   853→    conditions.push(eq(SessionTable.directory, input.directory))
   854→  }
   855→  if (input?.roots) {
   856→    conditions.push(isNull(SessionTable.parent_id))
   857→  }
   858→  if (input?.start) {
   859→    conditions.push(gte(SessionTable.time_updated, input.start))
   860→  }
   861→  if (input?.cursor) {
   862→    conditions.push(lt(SessionTable.time_updated, input.cursor))
   863→  }
   864→  if (input?.search) {
   865→    conditions.push(like(SessionTable.title, `%${input.search}%`))
   866→  }
   867→  if (!input?.archived) {
   868→    conditions.push(isNull(SessionTable.time_archived))
   869→  }
   870→
   871→  const limit = input?.limit ?? 100
   872→
   873→  const rows = Database.use((db) => {
   874→    const query =
   875→      conditions.length > 0
   876→        ? db
   877→            .select()
   878→            .from(SessionTable)
   879→            .where(and(...conditions))
   880→        : db.select().from(SessionTable)
   881→    return query.orderBy(desc(SessionTable.time_updated), desc(SessionTable.id)).limit(limit).all()
   882→  })
   883→
   884→  const ids = [...new Set(rows.map((row) => row.project_id))]
   885→  const projects = new Map<string, ProjectInfo>()
   886→
   887→  if (ids.length > 0) {
   888→    const items = Database.use((db) =>
   889→      db
   890→        .select({ id: ProjectTable.id, name: ProjectTable.name, worktree: ProjectTable.worktree })
   891→        .from(ProjectTable)
   892→        .where(inArray(ProjectTable.id, ids))
   893→        .all(),
   894→    )
   895→    for (const item of items) {
   896→      projects.set(item.id, {
   897→        id: item.id,
   898→        name: item.name ?? undefined,
   899→        worktree: item.worktree,
   900→      })
   901→    }
   902→  }
   903→
   904→  for (const row of rows) {
   905→    const project = projects.get(row.project_id) ?? null
   906→    yield { ...fromRow(row), project }
   907→  }
   908→}
